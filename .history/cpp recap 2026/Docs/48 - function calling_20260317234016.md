# 48 - Function Calling in C++

## What is Function Calling?
Function calling is the process of executing a function by using its name and passing arguments. When a function is called, program control jumps to that function, executes its code, and then returns to where it was called.

## Basic Function Call Syntax

```cpp
function_name(arguments);
```

## Simple Examples

```cpp
#include <iostream>
using namespace std;

// Function definitions
void sayHello() {
    cout << "Hello!" << endl;
}

int add(int a, int b) {
    return a + b;
}

int main() {
    // Function calls
    sayHello();  // Call void function
    
    int result = add(5, 3);  // Call function that returns value
    cout << "Result: " << result << endl;
    
    return 0;
}
```

## Calling Functions with Different Return Types

### Calling void Functions
```cpp
#include <iostream>
using namespace std;

void greet(string name) {
    cout << "Hello, " << name << "!" << endl;
}

void printLine() {
    cout << "--------------------" << endl;
}

int main() {
    // Direct calls - no return value to store
    greet("Alice");
    printLine();
    greet("Bob");
    
    // Can't use void function in expressions
    // int x = greet("John");  // ERROR!
    
    return 0;
}
```

### Calling Functions That Return Values
```cpp
#include <iostream>
using namespace std;

int getNumber() {
    return 42;
}

double calculateArea(double radius) {
    return 3.14159 * radius * radius;
}

bool isEven(int num) {
    return num % 2 == 0;
}

int main() {
    // Store return value
    int num = getNumber();
    cout << "Number: " << num << endl;
    
    // Use return value directly in expression
    double area = calculateArea(5.0);
    cout << "Area: " << area << endl;
    
    // Use in condition
    if (isEven(10)) {
        cout << "10 is even" << endl;
    }
    
    // Use in cout directly
    cout << "Another number: " << getNumber() << endl;
    
    return 0;
}
```

## Different Ways to Call Functions

### Function Call in Expression
```cpp
#include <iostream>
using namespace std;

int square(int x) {
    return x * x;
}

int main() {
    int result = square(5) + square(3);  // 25 + 9 = 34
    cout << "Result: " << result << endl;
    
    result = square(square(2));  // square(4) = 16
    cout << "Nested: " << result << endl;
    
    return 0;
}
```

### Function Call in Condition
```cpp
#include <iostream>
using namespace std;

bool isValid(int age) {
    return age >= 18 && age <= 65;
}

string getGrade(int score) {
    if (score >= 90) return "A";
    if (score >= 80) return "B";
    if (score >= 70) return "C";
    if (score >= 60) return "D";
    return "F";
}

int main() {
    int age = 25;
    
    // Function call in if condition
    if (isValid(age)) {
        cout << "Valid age for work" << endl;
    }
    
    // Function call in switch (not directly, but in assignment)
    char grade = getGrade(85)[0];  // First char of returned string
    
    // Function call in loop condition
    int count = 0;
    while (count < 5) {
        cout << "Count: " << count << endl;
        count++;
    }
    
    return 0;
}
```

### Nested Function Calls
```cpp
#include <iostream>
#include <cmath>
using namespace std;

int square(int x) {
    return x * x;
}

int cube(int x) {
    return x * x * x;
}

int main() {
    // Nested calls - functions called inside other functions
    int result = square(cube(2));  // cube(2) = 8, square(8) = 64
    cout << "square(cube(2)) = " << result << endl;
    
    result = cube(square(3));  // square(3) = 9, cube(9) = 729
    cout << "cube(square(3)) = " << result << endl;
    
    // Deep nesting
    double val = sqrt(abs(-25));  // abs(-25)=25, sqrt(25)=5
    cout << "sqrt(abs(-25)) = " << val << endl;
    
    return 0;
}
```

## Function Call Stack

```cpp
#include <iostream>
using namespace std;

void functionC() {
    cout << "In function C" << endl;
}

void functionB() {
    cout << "In function B, calling C" << endl;
    functionC();
    cout << "Back in function B" << endl;
}

void functionA() {
    cout << "In function A, calling B" << endl;
    functionB();
    cout << "Back in function A" << endl;
}

int main() {
    cout << "In main, calling A" << endl;
    functionA();
    cout << "Back in main" << endl;
    
    return 0;
}
```
Output:
```
In main, calling A
In function A, calling B
In function B, calling C
In function C
Back in function B
Back in function A
Back in main
```

## Calling Functions with Different Parameter Types

### Pass by Value
```cpp
#include <iostream>
using namespace std;

void increment(int x) {
    x++;  // Modifies local copy only
    cout << "Inside function: " << x << endl;
}

int main() {
    int num = 5;
    cout << "Before call: " << num << endl;
    
    increment(num);  // Passes a copy
    
    cout << "After call: " << num << endl;  // Still 5
    
    return 0;
}
```

### Pass by Reference
```cpp
#include <iostream>
using namespace std;

void increment(int &x) {
    x++;  // Modifies original
    cout << "Inside function: " << x << endl;
}

void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
    int num = 5;
    cout << "Before call: " << num << endl;
    
    increment(num);  // Passes reference
    
    cout << "After call: " << num << endl;  // Now 6
    
    int x = 10, y = 20;
    cout << "Before swap: x=" << x << ", y=" << y << endl;
    swap(x, y);
    cout << "After swap: x=" << x << ", y=" << y << endl;
    
    return 0;
}
```

### Pass by const Reference

#include <iostream>

### Passing Arrays
```cpp
#include <iostream>

void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {

    }

```

## Calling Functions from Other Functions

```cpp
#include <iostream>
#include <cmath>
using namespace std;

// Helper functions
double square(double x) {
    return x * x;
}

double cube(double x) {
    return x * x * x;
}

// Function that calls other functions
double power(double base, int exponent) {
    if (exponent == 0) return 1;
    if (exponent == 1) return base;
    if (exponent == 2) return square(base);
    if (exponent == 3) return cube(base);
    
    double result = 1;
    for (int i = 0; i < exponent; i++) {
        result *= base;
    }
    return result;
}

// Function that calls itself (recursive)
int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);  // Calls itself
}

int main() {
    cout << "2^2 = " << power(2, 2) << endl;
    cout << "2^3 = " << power(2, 3) << endl;
    cout << "2^5 = " << power(2, 5) << endl;
    cout << "Factorial 5 = " << factorial(5) << endl;
    
    return 0;
}
```

## Function Call with Default Arguments

```cpp
#include <iostream>
using namespace std;

// Function with default arguments
void display(string item, int count = 1, bool showPrice = false) {
    cout << count << " " << item;
    if (count > 1) cout << "s";
    if (showPrice) cout << " - $1.99 each";
    cout << endl;
}

int calculate(int a, int b = 1, int c = 1) {
    return a * b * c;
}

int main() {
    // Different ways to call with defaults
    display("apple");                    // Uses both defaults
    display("banana", 3);                // Uses showPrice default
    display("orange", 2, true);          // No defaults used
    
    cout << calculate(5) << endl;        // 5 * 1 * 1 = 5
    cout << calculate(5, 2) << endl;     // 5 * 2 * 1 = 10
    cout << calculate(5, 2, 3) << endl;  // 5 * 2 * 3 = 30
    
    return 0;
}
```

## Function Call with Overloaded Functions

```cpp
#include <iostream>
using namespace std;

// Overloaded functions
int add(int a, int b) {
    cout << "int version: ";
    return a + b;
}

double add(double a, double b) {
    cout << "double version: ";
    return a + b;
}

int add(int a, int b, int c) {
    cout << "three int version: ";
    return a + b + c;
}

int main() {
    // Compiler chooses correct version based on arguments
    cout << add(5, 3) << endl;        // Calls int version
    cout << add(2.5, 3.7) << endl;    // Calls double version
    cout << add(1, 2, 3) << endl;     // Calls three-int version
    
    return 0;
}
```

### Example 2: Student Grade System
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// Function definitions
double calculateAverage(const vector<int>& grades) {
    if (grades.empty()) return 0;
    int sum = 0;
    for (int grade : grades) {
        sum += grade;
    }
    return static_cast<double>(sum) / grades.size();
}

string getLetterGrade(double average) {
    if (average >= 90) return "A";
    if (average >= 80) return "B";
    if (average >= 70) return "C";
    if (average >= 60) return "D";
    return "F";
}

void displayResult(const string& name, double average, const string& grade) {
    cout << "\n=== Student Result ===" << endl;
    cout << "Name: " << name << endl;
    cout << "Average: " << average << endl;
    cout << "Grade: " << grade << endl;
}

int main() {
    string name;
    vector<int> grades;
    int grade;
    
    cout << "Enter student name: ";
    getline(cin, name);
    
    cout << "Enter grades (negative to stop):" << endl;
    while (true) {
        cout << "Grade: ";
        cin >> grade;
        if (grade < 0) break;
        grades.push_back(grade);
    }
    
    // Function calls
    double avg = calculateAverage(grades);
    string letterGrade = getLetterGrade(avg);
    displayResult(name, avg, letterGrade);
    
    return 0;
}
```

### Example 3: Temperature Converter

#include <iostream>

double celsiusToFahrenheit(double celsius) {
    return (celsius * 9.0/5.0) + 32;

    }
}

int main() {

### Example 4: Array Statistics
```cpp
#include <iostream>
#include <iomanip>
using namespace std;

void inputArray(int arr[], int size) {
    cout << "Enter " << size << " numbers:" << endl;
    for (int i = 0; i < size; i++) {
        cout << "arr[" << i << "]: ";
        cin >> arr[i];
    }
}

void printArray(const int arr[], int size) {
    cout << "[";
    for (int i = 0; i < size; i++) {
        cout << arr[i];
        if (i < size - 1) cout << ", ";
    }
    cout << "]" << endl;
}

int sumArray(const int arr[], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += arr[i];
    }
    return sum;
}

double averageArray(const int arr[], int size) {
    if (size == 0) return 0;
    return static_cast<double>(sumArray(arr, size)) / size;
}

int findMax(const int arr[], int size) {
    int max = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > max) max = arr[i];
    }
    return max;
}

int findMin(const int arr[], int size) {
    int min = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] < min) min = arr[i];
    }
    return min;
}

int main() {
    const int SIZE = 5;
    int numbers[SIZE];
    
    // Function calls
    inputArray(numbers, SIZE);
    
    cout << "\nArray: ";
    printArray(numbers, SIZE);
    
    cout << "Sum: " << sumArray(numbers, SIZE) << endl;
    cout << fixed << setprecision(2);
    cout << "Average: " << averageArray(numbers, SIZE) << endl;
    cout << "Maximum: " << findMax(numbers, SIZE) << endl;
    cout << "Minimum: " << findMin(numbers, SIZE) << endl;
    
    return 0;
}
```

## Common Mistakes in Function Calling

### Mistake 1: Wrong Number of Arguments
```cpp
int add(int a, int b) {
    return a + b;
}

int main() {
    // WRONG - too few arguments
    // int result = add(5);  // Error!
    
    // WRONG - too many arguments
    // int result = add(5, 3, 2);  // Error!
    
    // CORRECT
    int result = add(5, 3);
    
    return 0;
}
```

### Mistake 2: Wrong Argument Types
```cpp
void printPerson(string name, int age) {
    cout << name << " is " << age << " years old" << endl;
}

int main() {
    // WRONG - argument types mismatch
    // printPerson(25, "John");  // Error!
    
    // CORRECT
    printPerson("John", 25);
    
    return 0;
}
```

### Mistake 3: Forgetting Function Prototype
```cpp
// Missing prototype!

int main() {
    // WRONG - function not declared yet
    // sayHello();  // Error!
    
    return 0;
}

void sayHello() {
    cout << "Hello!" << endl;
}

// CORRECT - add prototype before main
void sayHello();  // Prototype
```

### Mistake 4: Ignoring Return Value
```cpp
int getNumber() {
    return 42;
}

int main() {
    // WRONG - ignoring return value (sometimes okay, but be careful)
    getNumber();  // Return value discarded
    
    // CORRECT - using return value
    int num = getNumber();
    cout << num << endl;
    
    return 0;
}
```

### Mistake 5: Calling Function Before Definition
```cpp
// WRONG - function called before definition
int main() {
    int result = add(5, 3);  // Error if no prototype
    return 0;
}

int add(int a, int b) {
    return a + b;
}

// CORRECT - with prototype
int add(int a, int b);  // Prototype

int main() {
    int result = add(5, 3);  // OK now
    return 0;
}

int add(int a, int b) {
    return a + b;
}
```

## Quick Reference

| Call Type | Syntax | Example |
|-----------|--------|---------|
| Void function | `func();` | `greet();` |
| With return | `var = func();` | `int x = getNum();` |
| With arguments | `func(arg1, arg2);` | `add(5, 3);` |
| Nested | `func1(func2(x));` | `sqrt(abs(-5));` |
| In expression | `x = func() + 5;` | `sum = add(a,b) * 2;` |
| In condition | `if (func()) { }` | `if (isValid(age)) { }` |

## Summary

- **Function call** transfers control to the function
- Arguments must match parameter types and order
- Return value can be stored, used in expressions, or ignored
- Functions can call other functions (including themselves)
- **Call stack** tracks nested function calls
- Pass by value copies data, pass by reference shares data
- Default arguments allow fewer arguments in call
- Overloaded functions are chosen based on arguments
- Always declare or define functions before calling them

## Basic Template
```cpp
#include <iostream>
using namespace std;

// Function definitions
int add(int a, int b) {
    return a + b;
}

void printResult(int x, int y, int result) {
    cout << x << " + " << y << " = " << result << endl;
}

int main() {
    int x = 5, y = 3;
    
    // Function calls
    int sum = add(x, y);           // Call with return
    printResult(x, y, sum);        // Call void function
    
    // Direct call in expression
    cout << "10 + 20 = " << add(10, 20) << endl;
    
    // Nested call
    cout << "add(add(1,2), add(3,4)) = " 
         << add(add(1, 2), add(3, 4)) << endl;
    
    return 0;
}
```
