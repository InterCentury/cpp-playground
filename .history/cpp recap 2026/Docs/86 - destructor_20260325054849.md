# 86 - Destructor in C++

## What is a Destructor?
A destructor is a special member function that is automatically called when an object goes out of scope or is explicitly deleted. It is used to release resources allocated by the object, such as dynamic memory, file handles, or network connections.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
using namespace std;

class Resource {
private:
    string name;
    int* data;
    int size;
    
public:
    // Constructor
    Resource(string n, int s) : name(n), size(s) {
        data = new int[size];
        cout << "Constructor: " << name << " allocated " << size << " ints" << endl;
    }
    
    // Destructor
    ~Resource() {
        delete[] data;
        cout << "Destructor: " << name << " released memory" << endl;
    }
    
    void display() const {
        cout << "Resource: " << name << " (size: " << size << ")" << endl;
    }
};

int main() {
    Resource r1("First", 100);
    Resource r2("Second", 200);
    
    r1.display();
    r2.display();
    
    // Destructors called automatically when objects go out of scope
    return 0;
}
```

## When Destructor is Called

```cpp
#include <iostream>
#include <string>
using namespace std;

class Demo {
private:
    string name;
    
public:
    Demo(string n) : name(n) {
        cout << "Constructor: " << name << endl;
    }
    
    ~Demo() {
        cout << "Destructor: " << name << endl;
    }
};

void function() {
    Demo local("Local in function");
    cout << "Inside function" << endl;
    // Destructor called when function exits
}

int main() {
    Demo d1("Global scope");
    
    {
        Demo d2("Block scope");
        cout << "Inside block" << endl;
    } // d2 destructor called here
    
    function();
    
    Demo* d3 = new Demo("Dynamic");
    delete d3;  // Destructor called explicitly
    
    cout << "End of main" << endl;
    // d1 destructor called here
    
    return 0;
}
```

## Practical Examples

### Example 1: File Handler with Destructor

#### Type 1: Single File
```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

class FileHandler {
private:
    string filename;
    fstream file;
    bool isOpen;
    
public:
    // Constructor - opens file
    FileHandler(string name, ios::openmode mode = ios::in | ios::out) 
        : filename(name), isOpen(false) {
        file.open(filename, mode);
        if (file.is_open()) {
            isOpen = true;
            cout << "File opened: " << filename << endl;
        } else {
            cout << "Failed to open: " << filename << endl;
        }
    }
    
    // Destructor - ensures file is closed
    ~FileHandler() {
        if (isOpen) {
            file.close();
            cout << "File closed: " << filename << endl;
        }
    }
    
    void write(const string& content) {
        if (isOpen) {
            file << content << endl;
            cout << "Written to: " << filename << endl;
        }
    }
    
    void read() {
        if (isOpen) {
            file.seekg(0);
            string line;
            while (getline(file, line)) {
                cout << line << endl;
            }
            file.clear(); // Clear EOF flag
        }
    }
};

int main() {
    FileHandler fh("test.txt");
    fh.write("Hello World");
    fh.write("Second line");
    fh.read();
    
    // Destructor automatically closes file
    return 0;
}
```

### Example 2: Dynamic Array Class with Destructor

#### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class IntArray {
private:
    int* data;
    int size;
    
public:
    // Constructor
    IntArray(int s) : size(s) {
        data = new int[size];
        for (int i = 0; i < size; i++) {
            data[i] = 0;
        }
        cout << "Array created (size " << size << ")" << endl;
    }
    
    // Destructor
    ~IntArray() {
        delete[] data;
        cout << "Array destroyed (size " << size << ")" << endl;
    }
    
    int& operator[](int index) {
        if (index >= 0 && index < size) {
            return data[index];
        }
        static int dummy = -1;
        return dummy;
    }
    
    int getSize() const { return size; }
    
    void display() const {
        for (int i = 0; i < size; i++) {
            cout << data[i] << " ";
        }
        cout << endl;
    }
};

int main() {
    IntArray arr1(5);
    IntArray arr2(10);
    
    arr1[0] = 10;
    arr1[1] = 20;
    
    cout << "arr1: ";
    arr1.display();
    
    cout << "arr2: ";
    arr2.display();
    
    // Destructors called automatically
    return 0;
}
```

### Example 3: Database Connection with Destructor

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
using namespace std;

class DatabaseConnection {
private:
    string connectionString;
    bool connected;
    
public:
    // Constructor
    DatabaseConnection(string connStr) : connectionString(connStr), connected(false) {
        cout << "Creating database connection object" << endl;
        connect();
    }
    
    // Destructor
    ~DatabaseConnection() {
        disconnect();
        cout << "Database connection object destroyed" << endl;
    }
    
    void connect() {
        if (!connected) {
            cout << "Connecting to: " << connectionString << endl;
            connected = true;
        }
    }
    
    void disconnect() {
        if (connected) {
            cout << "Disconnecting from: " << connectionString << endl;
            connected = false;
        }
    }
    
    void executeQuery(const string& query) {
        if (connected) {
            cout << "Executing: " << query << endl;
        } else {
            cout << "Not connected!" << endl;
        }
    }
};

int main() {
    {
        DatabaseConnection db("localhost:3306/mydb");
        db.executeQuery("SELECT * FROM users");
        db.executeQuery("INSERT INTO logs VALUES('test')");
    } // Destructor called - connection closed
    
    cout << "\nAfter scope" << endl;
    
    return 0;
}
```

## Virtual Destructor

```cpp
#include <iostream>
#include <string>
using namespace std;

class Base {
public:
    Base() { cout << "Base constructor" << endl; }
    
    // Virtual destructor - ensures proper cleanup of derived objects
    virtual ~Base() { cout << "Base destructor" << endl; }
};

class Derived : public Base {
private:
    int* data;
    
public:
    Derived() : data(new int[100]) {
        cout << "Derived constructor - allocated memory" << endl;
    }
    
    ~Derived() {
        delete[] data;
        cout << "Derived destructor - freed memory" << endl;
    }
};

int main() {
    cout << "=== Without virtual destructor, memory leak!" << endl;
    Base* ptr = new Derived();
    delete ptr;  // Calls Base destructor only if not virtual
    
    return 0;
}
```

## Rule of Three (Destructor, Copy Constructor, Assignment Operator)

```cpp
#include <iostream>
#include <cstring>
using namespace std;

class String {
private:
    char* data;
    int length;
    
public:
    // Constructor
    String(const char* s = "") {
        length = strlen(s);
        data = new char[length + 1];
        strcpy(data, s);
        cout << "Constructor: " << data << endl;
    }
    
    // Destructor
    ~String() {
        cout << "Destructor: " << data << endl;
        delete[] data;
    }
    
    // Copy constructor
    String(const String& other) {
        length = other.length;
        data = new char[length + 1];
        strcpy(data, other.data);
        cout << "Copy constructor: " << data << endl;
    }
    
    // Assignment operator
    String& operator=(const String& other) {
        if (this != &other) {
            delete[] data;
            length = other.length;
            data = new char[length + 1];
            strcpy(data, other.data);
            cout << "Assignment operator: " << data << endl;
        }
        return *this;
    }
    
    void display() const {
        cout << data << endl;
    }
};

int main() {
    String s1("Hello");
    String s2 = s1;      // Copy constructor
    String s3("World");
    s3 = s1;              // Assignment operator
    
    s1.display();
    s2.display();
    s3.display();
    
    return 0;
}
```

## Order of Destruction

```cpp
#include <iostream>
#include <string>
using namespace std;

class Member {
private:
    string name;
    
public:
    Member(string n) : name(n) {
        cout << "Member constructor: " << name << endl;
    }
    
    ~Member() {
        cout << "Member destructor: " << name << endl;
    }
};

class Container {
private:
    Member m1;
    Member m2;
    
public:
    Container() : m1("First"), m2("Second") {
        cout << "Container constructor" << endl;
    }
    
    ~Container() {
        cout << "Container destructor" << endl;
        // Members destroyed in reverse order of construction
    }
};

int main() {
    Container c;
    // Output shows destruction order: m2, m1, then container
    return 0;
}
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Missing delete | Memory leak | Always delete dynamically allocated memory |
| Double delete | Undefined behavior | Set pointer to nullptr after delete |
| Non-virtual destructor | Base class pointer, memory leak | Make destructor virtual in base class |
| Throwing exceptions in destructor | Program termination | Avoid exceptions in destructor |
| Accessing deleted memory | Crash | Don't use after delete |

## Summary

- **Destructor** cleans up resources when object is destroyed
- Named with `~` followed by class name
- No parameters, no return type
- Cannot be overloaded (only one destructor)
- Called automatically when object goes out of scope
- Called explicitly with `delete` for dynamic objects
- **Virtual destructor** essential for polymorphic base classes
- **Rule of Three**: if you need destructor, likely need copy constructor and assignment operator
- Destructors should **not throw exceptions**
