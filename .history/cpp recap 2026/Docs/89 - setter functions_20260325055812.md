# 89 - Setter Functions in C++

## What are Setter Functions?
Setter functions (also called mutators) are public member functions that allow controlled modification of private member variables. They provide validation, transformation, and encapsulation while maintaining data integrity.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
#include <algorithm>
#include <cctype>
using namespace std;

class Student {
private:
    string name;
    int rollNumber;
    double gpa;
    string email;
    
public:
    // Basic setters with validation
    void setName(const string& n) {
        if (n.length() >= 2 && n.length() <= 50) {
            name = n;
            // Capitalize first letter
            if (!name.empty()) {
                name[0] = toupper(name[0]);
            }
        } else {
            cout << "Invalid name length!" << endl;
        }
    }
    
    void setRollNumber(int roll) {
        if (roll > 0 && roll < 1000) {
            rollNumber = roll;
        } else {
            cout << "Invalid roll number!" << endl;
        }
    }
    
    void setGpa(double g) {
        if (g >= 0.0 && g <= 4.0) {
            gpa = g;
        } else {
            cout << "Invalid GPA! Must be between 0.0 and 4.0" << endl;
        }
    }
    
    void setEmail(const string& e) {
        // Basic email validation
        size_t atPos = e.find('@');
        size_t dotPos = e.find('.', atPos);
        if (atPos != string::npos && dotPos != string::npos && atPos > 0) {
            email = e;
        } else {
            cout << "Invalid email format!" << endl;
        }
    }
    
    // Getters for display
    string getName() const { return name; }
    int getRollNumber() const { return rollNumber; }
    double getGpa() const { return gpa; }
    string getEmail() const { return email; }
    
    void display() const {
        cout << "Name: " << name << endl;
        cout << "Roll: " << rollNumber << endl;
        cout << "GPA: " << gpa << endl;
        cout << "Email: " << email << endl;
    }
};

int main() {
    Student s;
    
    // Using setters
    s.setName("john doe");
    s.setRollNumber(101);
    s.setGpa(3.5);
    s.setEmail("john@example.com");
    
    s.display();
    
    // Invalid attempts
    s.setGpa(5.0);      // Invalid
    s.setRollNumber(-5); // Invalid
    s.setEmail("invalid"); // Invalid
    
    return 0;
}
```

### 2. Setter with Transformation
```cpp
class Name {
private:
    string firstName;
    string lastName;
    
public:
    void setFullName(const string& full) {
        size_t space = full.find(' ');
        if (space != string::npos) {
            firstName = full.substr(0, space);
            lastName = full.substr(space + 1);
            firstName[0] = toupper(firstName[0]);
            lastName[0] = toupper(lastName[0]);
        }
    }
};
```


### 4. Setter with Formatting
```cpp
class PhoneNumber {
private:
    string number;
    
public:
    void setNumber(const string& num) {
        // Remove all non-digit characters
        string cleaned;
        for (char c : num) {
            if (isdigit(c)) {
                cleaned += c;
            }
        }
        if (cleaned.length() == 10) {
            number = cleaned;
        }
    }
};
```

### 5. Fluent Setter (Method Chaining)
```cpp
class Person {
private:
    string name;
    int age;
    
public:
    Person& setName(const string& n) {
        name = n;
        return *this;
    }

        if (a >= 0) age = a;
        return *this;
    }
```

### Example 1: Product Class with Validation

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

class Product {
private:
    int id;
    string name;
    double price;
    int quantity;
    static int nextId;
    
public:
    Product() : id(nextId++) {}
    
    // Setters with validation
    void setName(const string& n) {
        if (n.length() >= 3 && n.length() <= 100) {
            name = n;
        } else {
            cout << "Product name must be 3-100 characters!" << endl;
        }
    }
    
    void setPrice(double p) {
        if (p >= 0) {
            price = p;
        } else {
            cout << "Price cannot be negative!" << endl;
        }
    }
    
    void setQuantity(int q) {
        if (q >= 0) {
            quantity = q;
        } else {
            cout << "Quantity cannot be negative!" << endl;
        }
    }
    
    void setId(int i) {
        if (i > 0) {
            id = i;
        }
    }
    
    // Setters with calculations
    void applyDiscount(double percent) {
        if (percent >= 0 && percent <= 100) {
            price = price * (1 - percent / 100);
            cout << "Discount applied! New price: $" << price << endl;
        }
    }
    
    void restock(int amount) {
        if (amount > 0) {
            quantity += amount;
            cout << "Added " << amount << " units. New stock: " << quantity << endl;
        }
    }
    
    void sell(int amount) {
        if (amount > 0 && amount <= quantity) {
            quantity -= amount;
            cout << "Sold " << amount << " units. Remaining: " << quantity << endl;
        } else {
            cout << "Insufficient stock!" << endl;
        }
    }
    
    // Getters
    int getId() const { return id; }
    string getName() const { return name; }
    double getPrice() const { return price; }
    int getQuantity() const { return quantity; }
    double getTotalValue() const { return price * quantity; }
    
    void display() const {
        cout << fixed << setprecision(2);
        cout << "ID: " << id << ", Name: " << name << endl;
        cout << "Price: $" << price << ", Stock: " << quantity << endl;
        cout << "Total Value: $" << getTotalValue() << endl;
    }
};

int Product::nextId = 1000;

int main() {
    Product p1, p2;
    
    p1.setName("Laptop");
    p1.setPrice(999.99);
    p1.setQuantity(10);
    
    p2.setName("Mouse");
    p2.setPrice(29.99);
    p2.setQuantity(50);
    
    p1.display();
    cout << endl;
    p2.display();
    
    p1.applyDiscount(10);
    p1.sell(2);
    p1.restock(5);
    
    cout << endl;
    p1.display();
    
    return 0;
}
```

### Example 2: Date Class with Validation

#### Type 1: Single File
```cpp
#include <iostream>
#include <iomanip>
using namespace std;

class Date {
private:
    int day;
    int month;
    int year;
    
    bool isLeapYear() const {
        return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
    }
    
    int daysInMonth() const {
        if (month == 2) return isLeapYear() ? 29 : 28;
        if (month == 4 || month == 6 || month == 9 || month == 11) return 30;
        return 31;
    }
    
    bool isValidDate() const {
        if (year < 1900 || year > 2100) return false;
        if (month < 1 || month > 12) return false;
        if (day < 1 || day > daysInMonth()) return false;
        return true;
    }
    
public:
    // Setters with validation
    void setDay(int d) {
        int oldDay = day;
        day = d;
        if (!isValidDate()) {
            day = oldDay;
            cout << "Invalid day! Keeping previous value." << endl;
        }
    }
    
    void setMonth(int m) {
        int oldMonth = month;
        month = m;
        if (!isValidDate()) {
            month = oldMonth;
            cout << "Invalid month! Keeping previous value." << endl;
        }
    }
    
    void setYear(int y) {
        int oldYear = year;
        year = y;
        if (!isValidDate()) {
            year = oldYear;
            cout << "Invalid year! Keeping previous value." << endl;
        }
    }
    
    void setDate(int d, int m, int y) {
        day = d;
        month = m;
        year = y;
        if (!isValidDate()) {
            cout << "Invalid date! Setting to 1/1/2000" << endl;
            day = 1;
            month = 1;
            year = 2000;
        }
    }
    
    // Setters with relative changes
    void addDays(int days) {
        day += days;
        while (day > daysInMonth()) {
            day -= daysInMonth();
            month++;
            if (month > 12) {
                month = 1;
                year++;
            }
        }
        while (day < 1) {
            month--;
            if (month < 1) {
                month = 12;
                year--;
            }
            day += daysInMonth();
        }
    }
    
    // Getters
    int getDay() const { return day; }
    int getMonth() const { return month; }
    int getYear() const { return year; }
    
    void display() const {
        cout << setfill('0') << setw(2) << day << "/"
             << setw(2) << month << "/" << year;
    }
};

int main() {
    Date d;
    
    d.setDate(15, 3, 2024);
    cout << "Date: "; d.display(); cout << endl;
    
    d.setDay(32);  // Invalid
    d.setMonth(13); // Invalid
    d.setYear(3000); // Invalid
    
    d.addDays(10);
    cout << "After +10 days: "; d.display(); cout << endl;
    
    d.addDays(-20);
    cout << "After -20 days: "; d.display(); cout << endl;
    
    return 0;
}
```

### Example 3: Fluent Setter Interface (Method Chaining)

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

class Employee {
private:
    string name;
    string title;
    double salary;
    string department;
    
public:
    // Fluent setters - return reference to *this
    Employee& setName(const string& n) {
        name = n;
        if (!name.empty()) {
            name[0] = toupper(name[0]);
        }
        return *this;
    }
    
    Employee& setTitle(const string& t) {
        title = t;
        return *this;
    }
    
    Employee& setSalary(double s) {
        if (s >= 0) {
            salary = s;
        }
        return *this;
    }
    
    Employee& setDepartment(const string& d) {
        department = d;
        return *this;
    }
    
    // Batch setters
    Employee& setAll(const string& n, const string& t, double s, const string& d) {
        setName(n).setTitle(t).setSalary(s).setDepartment(d);
        return *this;
    }
    
    void display() const {
        cout << "Name: " << name << endl;
        cout << "Title: " << title << endl;
        cout << "Salary: $" << salary << endl;
        cout << "Department: " << department << endl;
    }
};

int main() {
    Employee e;
    
    // Method chaining - clean and readable
    e.setName("john doe")
     .setTitle("Software Engineer")
     .setSalary(75000)
     .setDepartment("Engineering");
    
    e.display();
    
    cout << endl;
    
    // Batch setting
    Employee e2;
    e2.setAll("jane smith", "Project Manager", 90000, "Management");
    e2.display();
    
    return 0;
}
```

## Common Setter Patterns

```cpp
class Configuration {
private:
    string value;
    bool modified;
    time_t lastModified;
    
public:
    // Basic setter with validation
    void setValue(const string& v) {
        if (!v.empty()) {
            value = v;
            modified = true;
            lastModified = time(nullptr);
        }
    }
    
    // Setter with transformation
    void setLowercase(const string& v) {
        value = v;
        for (char& c : value) {
            c = tolower(c);
        }
    }
    
    // Setter with side effect
    void setAndNotify(const string& v) {
        value = v;
        cout << "Configuration changed to: " << value << endl;
    }
    
    // Conditional setter
    bool trySetValue(const string& v) {
        if (isValid(v)) {
            value = v;
            return true;
        }
        return false;
    }
    
    bool isValid(const string& v) const {
        return v.length() >= 3;
    }
};
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| No validation | Invalid data can be set | Always validate in setters |
| Forgetting to update related members | Inconsistent state | Update all affected members |
| Returning void | Cannot chain | Return reference to *this for chaining |
| Exposing internal representation | Encapsulation broken | Return const reference or copy |
| Not handling edge cases | Unexpected behavior | Validate all inputs |

## Summary

- **Setters** provide controlled modification of private members
- **Validation** ensures data integrity
- **Transformation** can format/clean data before storing
- **Side effects** can update related members or timestamps
- **Fluent setters** enable method chaining for cleaner code
- **Batch setters** allow setting multiple attributes at once
- Always maintain **invariants** and **consistency**
