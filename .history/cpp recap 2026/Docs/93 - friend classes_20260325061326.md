# 93 - Friend Classes in C++

## What are Friend Classes?
A friend class is a class that is granted access to the private and protected members of another class. When Class A declares Class B as a friend, Class B can access all members of Class A, including private ones.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    string accountNumber;
    double balance;
    string pin;
    
public:
    BankAccount(string accNum, double initial, string p) 
        : accountNumber(accNum), balance(initial), pin(p) {}
    
    // Declare AccountManager as friend class
    friend class AccountManager;
    
    void display() const {
        cout << "Account: " << accountNumber << ", Balance: $" << balance << endl;
    }
};

class AccountManager {
public:
    void deposit(BankAccount& acc, double amount) {
        acc.balance += amount;  // Direct access to private member
        cout << "Deposited $" << amount << " to " << acc.accountNumber << endl;
    }
    
    bool withdraw(BankAccount& acc, double amount, const string& inputPin) {
        if (inputPin == acc.pin && amount <= acc.balance) {
            acc.balance -= amount;
            cout << "Withdrawn $" << amount << " from " << acc.accountNumber << endl;
            return true;
        }
        return false;
    }
    
    double getBalance(const BankAccount& acc) {
        return acc.balance;  // Direct access to private member
    }
    
    void transfer(BankAccount& from, BankAccount& to, double amount, const string& pin) {
        if (withdraw(from, amount, pin)) {
            deposit(to, amount);
            cout << "Transferred $" << amount << " from " << from.accountNumber 
                 << " to " << to.accountNumber << endl;
        }
    }
};

int main() {
    BankAccount acc1("ACC123", 1000, "1234");
    BankAccount acc2("ACC456", 500, "5678");
    AccountManager manager;
    
    acc1.display();
    acc2.display();
    
    manager.deposit(acc1, 500);
    manager.withdraw(acc2, 200, "5678");
    
    manager.transfer(acc1, acc2, 300, "1234");
    
    cout << "\nFinal Balances:" << endl;
    acc1.display();
    acc2.display();
    
    return 0;
}
```

## Friend Classes in Different Scenarios

### 1. Bidirectional Friendship
```cpp
#include <iostream>
#include <string>
using namespace std;

class ClassA;  // Forward declaration

class ClassB {
private:
    int secretB;
    
public:
    ClassB(int val) : secretB(val) {}
    
    // Friend function to access A's private members
    friend void exchangeSecrets(ClassA& a, ClassB& b);
};

class ClassA {
private:
    int secretA;
    
public:
    ClassA(int val) : secretA(val) {}
    
    // Friend function to access B's private members
    friend void exchangeSecrets(ClassA& a, ClassB& b);
};

void exchangeSecrets(ClassA& a, ClassB& b) {
    cout << "A's secret: " << a.secretA << endl;
    cout << "B's secret: " << b.secretB << endl;
    
    // Swap secrets
    int temp = a.secretA;
    a.secretA = b.secretB;
    b.secretB = temp;
    
    cout << "After swap:" << endl;
    cout << "A's secret: " << a.secretA << endl;
    cout << "B's secret: " << b.secretB << endl;
}

int main() {
    ClassA a(100);
    ClassB b(200);
    
    exchangeSecrets(a, b);
    
    return 0;
}
```

### 2. Nested Classes as Friends
```cpp
#include <iostream>
#include <string>
using namespace std;

class Container {
private:
    string secret;
    int data[10];
    
public:
    Container() : secret("Top Secret") {
        for (int i = 0; i < 10; i++) data[i] = i * 10;
    }
    
    // Nested class that is a friend
    class Iterator {
    private:
        Container* container;
        int index;
        
    public:
        Iterator(Container* c) : container(c), index(0) {}
        
        bool hasNext() const { return index < 10; }
        
        int next() {
            return container->data[index++];  // Access private member
        }
        
        void revealSecret() {
            cout << "Secret: " << container->secret << endl;  // Access private
        }
    };
    
    // Make nested class a friend
    friend class Iterator;
};

int main() {
    Container c;
    Container::Iterator it(&c);
    
    it.revealSecret();
    
    cout << "Data: ";
    while (it.hasNext()) {
        cout << it.next() << " ";
    }
    cout << endl;
    
    return 0;
}
```

### Example 1: Linked List with Node and List Classes

#### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class LinkedList;  // Forward declaration

class Node {
private:
    int data;
    Node* next;
    
    // Node constructor is private - only LinkedList can create nodes
    Node(int val) : data(val), next(nullptr) {}
    
    friend class LinkedList;
};

class LinkedList {
private:
    Node* head;
    
public:
    LinkedList() : head(nullptr) {}
    
    void insert(int val) {
        Node* newNode = new Node(val);  // Can create Node due to friendship
        newNode->next = head;
        head = newNode;
        cout << "Inserted: " << val << endl;
    }
    
    void display() const {
        Node* current = head;
        while (current) {
            cout << current->data << " -> ";
            current = current->next;
        }
        cout << "NULL" << endl;
    }
    
    bool search(int val) const {
        Node* current = head;
        while (current) {
            if (current->data == val) return true;
            current = current->next;
        }
        return false;
    }
    
    ~LinkedList() {
        Node* current = head;
        while (current) {
            Node* temp = current;
            current = current->next;
            delete temp;
        }
    }
};

int main() {
    LinkedList list;
    
    list.insert(10);
    list.insert(20);
    list.insert(30);
    
    list.display();
    
    cout << "Search 20: " << (list.search(20) ? "Found" : "Not found") << endl;
    cout << "Search 40: " << (list.search(40) ? "Found" : "Not found") << endl;
    
    return 0;
}
```

### Example 2: Builder Pattern with Friend Class

```cpp

#include <vector>

    string ram;
```

### Example 3: Database Connection Pool with Friend Classes

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <mutex>
using namespace std;

class DatabaseConnection {
private:
    string connectionString;
    bool inUse;
    
    DatabaseConnection(const string& conn) : connectionString(conn), inUse(false) {}
    
    void connect() {
        cout << "Connected to " << connectionString << endl;
    }
    
    void disconnect() {
        cout << "Disconnected from " << connectionString << endl;
    }
    
    friend class ConnectionPool;
};

class ConnectionPool {
private:
    vector<DatabaseConnection*> connections;
    string connectionString;
    int maxConnections;
    mutex mtx;
    
public:
    ConnectionPool(const string& conn, int max) 
        : connectionString(conn), maxConnections(max) {}
    
    DatabaseConnection* acquire() {
        lock_guard<mutex> lock(mtx);
        
        // Find free connection
        for (auto conn : connections) {
            if (!conn->inUse) {
                conn->inUse = true;
                conn->connect();
                return conn;
            }
        }
        
        // Create new connection if pool not full
        if (connections.size() < maxConnections) {
            DatabaseConnection* newConn = new DatabaseConnection(connectionString);
            newConn->inUse = true;
            newConn->connect();
            connections.push_back(newConn);
            return newConn;
        }
        
        cout << "No available connections!" << endl;
        return nullptr;
    }
    
    void release(DatabaseConnection* conn) {
        lock_guard<mutex> lock(mtx);
        if (conn) {
            conn->disconnect();
            conn->inUse = false;
            cout << "Connection returned to pool" << endl;
        }
    }
    
    ~ConnectionPool() {
        for (auto conn : connections) {
            delete conn;
        }
    }
};

int main() {
    ConnectionPool pool("localhost:3306/mydb", 2);
    
    DatabaseConnection* conn1 = pool.acquire();
    DatabaseConnection* conn2 = pool.acquire();
    DatabaseConnection* conn3 = pool.acquire();  // Should wait or fail
    
    pool.release(conn1);
    
    DatabaseConnection* conn4 = pool.acquire();  // Now available
    
    pool.release(conn2);
    pool.release(conn4);
    
    return 0;
}
```

## Friend Class vs Inheritance

```cpp
#include <iostream>
#include <string>
using namespace std;

// Friend Class Approach
class FriendApproach {
private:
    int secret;
    
public:
    FriendApproach(int s) : secret(s) {}
    
    friend class FriendAccessor;
};

class FriendAccessor {
public:
    void reveal(const FriendApproach& obj) {
        cout << "Friend class can access: " << obj.secret << endl;
    }
};

// Inheritance Approach
class Base {
protected:
    int secret;
    
public:
    Base(int s) : secret(s) {}
};

class Derived : public Base {
public:
    Derived(int s) : Base(s) {}
    
    void reveal() {
        cout << "Derived class can access: " << secret << endl;
    }
};

int main() {
    FriendApproach fa(100);
    FriendAccessor acc;
    acc.reveal(fa);
    
    Derived d(200);
    d.reveal();
    
    return 0;
}
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Forgetting forward declaration | Compilation error | Declare class before use |
| Mutual friendship without forward decl | Circular dependency | Use forward declaration |
| Overusing friendship | Breaks encapsulation | Use only when necessary |
| Friend class accessing invalid members | Runtime error | Ensure members exist |
| Friendship not transitive | Unexpected access | Friend of friend has no access |

## Friendship Properties

```cpp
#include <iostream>
using namespace std;

class A {
private:
    int secretA = 10;
    
    friend class B;  // B can access A
};

class B {
private:
    int secretB = 20;
    
    friend class C;  // C can access B
    // C cannot access A (friendship is not transitive)
};

class C {
public:
    void test() {
        A a;
        // cout << a.secretA;  // Error! C cannot access A
        
        B b;
        cout << "B's secret: " << b.secretB << endl;  // OK - C is friend of B
    }
};

int main() {
    C c;
    c.test();
    return 0;
}
```

## Quick Reference

| Feature | Friend Function | Friend Class |
|---------|-----------------|--------------|
| Access | All members of declaring class | All members of declaring class |
| Declaration | `friend void func();` | `friend class ClassName;` |
| Scope | Not a member | Not a member |
| Multiple classes | Yes | Yes |
| Transitive | No | No |
| Inheritance | Not inherited | Not inherited |

## Summary

- **Friend class** can access private/protected members of another class
- Declared with `friend class ClassName;` inside the granting class
- **Not transitive** (friend of friend is not friend)
- **Not inherited** (derived classes don't inherit friendship)
- Useful for **tightly coupled classes** (iterator, builder, manager)
- **Breaks encapsulation** - use sparingly
- Common in **container classes**, **builder pattern**, **unit testing**
