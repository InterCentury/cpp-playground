# 76 - Member Functions in C++

## What are Member Functions?
Member functions (also called methods) are functions that belong to a class. They define the behavior and operations that can be performed on objects of that class. Member functions have access to the class's private and protected members.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
#include <cmath>
using namespace std;

class Circle {
private:
    double radius;
    
public:
    // Member functions defined inside class
    void setRadius(double r) {
        if (r >= 0) radius = r;
    }
    
    double getRadius() const { return radius; }
    double area() const { return 3.14159 * radius * radius; }
    double circumference() const { return 2 * 3.14159 * radius; }
    
    void display() const {
        cout << "Radius: " << radius << ", Area: " << area() 
             << ", Circumference: " << circumference() << endl;
    }
};

int main() {
    Circle c;
    c.setRadius(5.0);
    c.display();
    
    return 0;
}
```

## Types of Member Functions

### 1. Accessor (Getter) Functions
```cpp
class BankAccount {
private:
    double balance;
    
public:
    double getBalance() const { return balance; }  // Accessor
    void deposit(double amount) {                  // Mutator
        if (amount > 0) balance += amount;
    }
};
```

### 2. Const Member Functions
```cpp
class Person {
private:
    string name;
    
public:
    string getName() const { return name; }     // Can't modify object
    void setName(string n) { name = n; }        // Can modify
};
```

### 3. Static Member Functions
```cpp
class Student {
private:
    static int count;
    
public:
    Student() { count++; }
    static int getCount() { return count; }     // Called without object
};

int Student::count = 0;
// Usage: Student::getCount()
```

### 4. Inline Member Functions
```cpp
class Math {
public:
    // Implicitly inline (defined inside class)
    int square(int x) { return x * x; }
    
    // Explicitly inline (defined outside)
    inline int cube(int x);
};

inline int Math::cube(int x) { return x * x * x; }
```

## Special Member Functions

### Constructor
```cpp
class Book {
private:
    string title;
    int pages;
    
public:
    // Default constructor
    Book() : title("Unknown"), pages(0) {}
    
    // Parameterized constructor
    Book(string t, int p) : title(t), pages(p) {}
    
    // Copy constructor
    Book(const Book& other) : title(other.title), pages(other.pages) {}
    
    void display() const {
        cout << title << ", " << pages << " pages" << endl;
    }
};
```

### Destructor
```cpp
class Resource {
private:
    int* data;
    int size;
    
public:
    Resource(int s) : size(s) {
        data = new int[size];
    }
    
    ~Resource() {
        delete[] data;  // Clean up dynamically allocated memory
    }
};
```

## Operator Overloading as Member Functions

```cpp
class Complex {
private:
    double real, imag;
    
public:
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}
    
    Complex operator+(const Complex& other) const {
        return Complex(real + other.real, imag + other.imag);
    }
    
    Complex operator-(const Complex& other) const {
        return Complex(real - other.real, imag - other.imag);
    }
    
    bool operator==(const Complex& other) const {
        return real == other.real && imag == other.imag;
    }
    
    void display() const {
        cout << real << " + " << imag << "i" << endl;
    }
};

int main() {
    Complex c1(3, 4), c2(1, 2);
    Complex sum = c1 + c2;
    sum.display();
    return 0;
}
```

## Practical Examples

### Example 1: Stack Class

#### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class Stack {
private:
    static const int MAX = 100;
    int arr[MAX];
    int top;
    
public:
    Stack() : top(-1) {}
    
    bool isEmpty() const { return top == -1; }
    bool isFull() const { return top == MAX - 1; }
    
    bool push(int value) {
        if (isFull()) return false;
        arr[++top] = value;
        return true;
    }
    
    bool pop(int& value) {
        if (isEmpty()) return false;
        value = arr[top--];
        return true;
    }
    
    int peek() const {
        if (isEmpty()) return -1;
        return arr[top];
    }
    
    void display() const {
        for (int i = top; i >= 0; i--)
            cout << arr[i] << " ";
        cout << endl;
    }
};

int main() {
    Stack s;
    s.push(10);
    s.push(20);
    s.push(30);
    s.display();
    
    int val;
    s.pop(val);
    cout << "Popped: " << val << endl;
    s.display();
    cout << "Top: " << s.peek() << endl;
    
    return 0;
}
```

### Example 2: Bank Account with Transaction History

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <ctime>
using namespace std;

class Transaction {
private:
    string type;
    double amount;
    time_t timestamp;
    
public:
    Transaction(string t, double a) : type(t), amount(a) {
        timestamp = time(nullptr);
    }
    
    void display() const {
        cout << "  " << ctime(&timestamp);
        cout << "  " << type << ": $" << amount << endl;
    }
};

class BankAccount {
private:
    string accountNumber;
    string accountHolder;
    double balance;
    vector<Transaction> history;
    
public:
    BankAccount(string number, string holder, double initial = 0) 
        : accountNumber(number), accountHolder(holder), balance(initial) {
        if (initial > 0) {
            history.push_back(Transaction("Initial deposit", initial));
        }
    }
    
    bool deposit(double amount) {
        if (amount <= 0) return false;
        balance += amount;
        history.push_back(Transaction("Deposit", amount));
        return true;
    }
    
    bool withdraw(double amount) {
        if (amount <= 0 || amount > balance) return false;
        balance -= amount;
        history.push_back(Transaction("Withdrawal", amount));
        return true;
    }
    
    double getBalance() const { return balance; }
    string getAccountNumber() const { return accountNumber; }
    string getAccountHolder() const { return accountHolder; }
    
    void displayStatement() const {
        cout << "\n=== Account Statement ===" << endl;
        cout << "Account: " << accountNumber << endl;
        cout << "Holder: " << accountHolder << endl;
        cout << "Balance: $" << balance << endl;
        cout << "\nTransaction History:" << endl;
        
        for (const auto& t : history) {
            t.display();
        }
    }
};

int main() {
    BankAccount acc("123456", "John Doe", 1000);
    acc.deposit(500);
    acc.withdraw(200);
    acc.deposit(300);
    acc.displayStatement();
    
    return 0;
}
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Forgetting definition | Linker error | Provide function definition |
| Missing const | Can't call on const objects | Add `const` to getters |
| Ambiguous parameters | Parameter shadows member | Use `this->` or different names |
| Returning local reference | Dangling reference | Return by value or member reference |
| Non-const in const function | Compilation error | Make function const or remove const |

## Quick Reference Table

| Member Function Type | Syntax | Key Feature |
|---------------------|--------|-------------|
| Accessor | `int getX() const` | Returns value, read-only |
| Mutator | `void setX(int x)` | Modifies member |
| Const | `void display() const` | Cannot modify object |
| Static | `static int count()` | Called without object |
| Inline | `inline int square(int x)` | Suggests code insertion |
| Virtual | `virtual void draw()` | Polymorphism |
| Constructor | `ClassName()` | Initializes object |
| Destructor | `~ClassName()` | Cleans up resources |

## Summary

- **Member functions** define behavior for class objects
- **Accessors** (getters) return member values with `const`
- **Mutators** (setters) modify member values with validation
- **Const member functions** cannot modify object state
- **Static member functions** belong to class, called via `ClassName::`
- **Constructors** initialize objects; **destructors** clean up
- **Inline functions** suggest compiler to avoid call overhead
- **Operator overloading** provides intuitive syntax for objects

## Basic Template

### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class MyClass {
private:
    int value;
    static int count;
    
public:
    MyClass(int v = 0) : value(v) { count++; }
    ~MyClass() { count--; }
    
    int getValue() const { return value; }
    void setValue(int v) { value = v; }
    static int getCount() { return count; }
    
    void display() const {
        cout << "Value: " << value << endl;
    }
};

int MyClass::count = 0;

int main() {
    MyClass obj1(10), obj2(20);
    obj1.display();
    cout << "Objects: " << MyClass::getCount() << endl;
    return 0;
}
```
