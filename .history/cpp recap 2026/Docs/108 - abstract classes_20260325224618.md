# 108 - Abstract Classes in C++

## What are Abstract Classes?
An abstract class is a class that cannot be instantiated. It serves as a blueprint for derived classes, defining an interface that derived classes must implement. A class becomes abstract when it contains at least one pure virtual function.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
#include <cmath>
using namespace std;

// Abstract class - cannot be instantiated
class Shape {
protected:
    string color;
    
public:
    Shape(string c) : color(c) {
        cout << "Shape constructor called" << endl;
    }
    
    // Pure virtual functions - make class abstract
    virtual double area() const = 0;
    virtual double perimeter() const = 0;
    virtual void draw() const = 0;
    
    // Concrete virtual function - has implementation
    virtual void display() const {
        cout << "Shape color: " << color << endl;
        cout << "Area: " << area() << endl;
        cout << "Perimeter: " << perimeter() << endl;
    }
    
    string getColor() const { return color; }
    
    // Virtual destructor
    virtual ~Shape() {
        cout << "Shape destructor called" << endl;
    }
};

// Concrete class - implements all pure virtual functions
class Circle : public Shape {
private:
    double radius;
    
public:
    Circle(string c, double r) : Shape(c), radius(r) {
        cout << "Circle constructor called" << endl;
    }
    
    double area() const override {
        return M_PI * radius * radius;
    }
    
    double perimeter() const override {
        return 2 * M_PI * radius;
    }
    
    void draw() const override {
        cout << "○ Drawing a circle with radius " << radius << endl;
    }
    
    ~Circle() {
        cout << "Circle destructor called" << endl;
    }
};

class Rectangle : public Shape {
private:
    double length;
    double width;
    
public:
    Rectangle(string c, double l, double w) : Shape(c), length(l), width(w) {
        cout << "Rectangle constructor called" << endl;
    }
    
    double area() const override {
        return length * width;
    }
    
    double perimeter() const override {
        return 2 * (length + width);
    }
    
    void draw() const override {
        cout << "□ Drawing a rectangle " << length << " x " << width << endl;
    }
    
    ~Rectangle() {
        cout << "Rectangle destructor called" << endl;
    }
};

int main() {
    // Shape s;  // Error! Cannot instantiate abstract class
    
    // Can use pointers and references to abstract class
    Shape* shapes[2];
    shapes[0] = new Circle("Red", 5.0);
    shapes[1] = new Rectangle("Blue", 4.0, 3.0);
    
    for (int i = 0; i < 2; i++) {
        shapes[i]->draw();
        shapes[i]->display();
        cout << endl;
        delete shapes[i];
    }
    
    return 0;
}
```

## Abstract Class as Interface

```cpp
#include <iostream>
#include <string>
using namespace std;

// Interface for Drawable objects
class Drawable {
public:
    virtual void draw() const = 0;
    virtual void setColor(const string& color) = 0;
    virtual ~Drawable() {}
};

// Interface for Printable objects
class Printable {
public:
    virtual void print() const = 0;
    virtual void setFont(const string& font) = 0;
    virtual ~Printable() {}
};

// Interface for Serializable objects
class Serializable {
public:
    virtual void serialize() const = 0;
    virtual void deserialize(const string& data) = 0;
    virtual ~Serializable() {}
};

// Concrete class implementing multiple interfaces
class Document : public Drawable, public Printable, public Serializable {
private:
    string title;
    string content;
    string color;
    string font;
    
public:
    Document(string t, string c) : title(t), content(c), color("black"), font("Arial") {}
    
    // Drawable implementation
    void draw() const override {
        cout << "Drawing document: " << title << " in " << color << endl;
    }
    
    void setColor(const string& c) override {
        color = c;
        cout << "Color set to: " << color << endl;
    }
    
    // Printable implementation
    void print() const override {
        cout << "=== " << title << " ===" << endl;
        cout << content << endl;
        cout << "Font: " << font << endl;
    }
    
    void setFont(const string& f) override {
        font = f;
        cout << "Font set to: " << font << endl;
    }
    
    // Serializable implementation
    void serialize() const override {
        cout << "<doc><title>" << title << "</title>" << endl;
        cout << "<content>" << content << "</content></doc>" << endl;
    }
    
    void deserialize(const string& data) override {
        cout << "Deserializing: " << data << endl;
    }
};

int main() {
    Document doc("MyDoc", "Hello World");
    
    Drawable* drawable = &doc;
    Printable* printable = &doc;
    Serializable* serializable = &doc;
    
    drawable->setColor("Blue");
    drawable->draw();
    cout << endl;
    
    printable->setFont("Times New Roman");
    printable->print();
    cout << endl;
    
    serializable->serialize();
    
    return 0;
}
```

## Practical Examples

### Example 1: Employee Management System with Abstract Base Class

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// Abstract base class
class Employee {
protected:
    int id;
    string name;
    double baseSalary;
    static int nextId;
    
public:
    Employee(string n, double salary) : name(n), baseSalary(salary) {
        id = nextId++;
        cout << "Employee created: " << name << " (ID: " << id << ")" << endl;
    }
    
    // Pure virtual functions
    virtual double calculateSalary() const = 0;
    virtual string getRole() const = 0;
    virtual void performDuties() const = 0;
    
    // Concrete methods
    virtual void display() const {
        cout << "ID: " << id << ", Name: " << name << endl;
        cout << "Role: " << getRole() << endl;
        cout << "Salary: $" << calculateSalary() << endl;
    }
    
    int getId() const { return id; }
    string getName() const { return name; }
    
    virtual ~Employee() {
        cout << "Employee destroyed: " << name << endl;
    }
};

int Employee::nextId = 1000;

class Developer : public Employee {
private:
    string programmingLanguage;
    int projectsCompleted;
    
public:
    Developer(string n, double salary, string lang, int projects)
        : Employee(n, salary), programmingLanguage(lang), projectsCompleted(projects) {
        cout << "  Developer: " << lang << " specialist" << endl;
    }
    
    double calculateSalary() const override {
        return baseSalary + (projectsCompleted * 1000);
    }
    
    string getRole() const override {
        return "Software Developer";
    }
    
    void performDuties() const override {
        cout << name << " is writing " << programmingLanguage << " code" << endl;
        cout << name << " is debugging applications" << endl;
        cout << name << " is attending code reviews" << endl;
    }
    
    void display() const override {
        Employee::display();
        cout << "Language: " << programmingLanguage << endl;
        cout << "Projects Completed: " << projectsCompleted << endl;
    }
    
    ~Developer() {
        cout << "  Developer destroyed" << endl;
    }
};

class Manager : public Employee {
private:
    int teamSize;
    vector<string> teamMembers;
    
public:
    Manager(string n, double salary, int size) : Employee(n, salary), teamSize(size) {
        cout << "  Manager: leading " << size << " people" << endl;
    }
    
    void addTeamMember(string member) {
        teamMembers.push_back(member);
    }
    
    double calculateSalary() const override {
        return baseSalary + (teamSize * 500);
    }
    
    string getRole() const override {
        return "Team Manager";
    }
    
    void performDuties() const override {
        cout << name << " is conducting team meetings" << endl;
        cout << name << " is reviewing code" << endl;
        cout << name << " is planning sprints" << endl;
        cout << name << " is mentoring " << teamSize << " developers" << endl;
    }
    
    void display() const override {
        Employee::display();
        cout << "Team Size: " << teamSize << endl;
        cout << "Team Members: ";
        for (const auto& m : teamMembers) {
            cout << m << " ";
        }
        cout << endl;
    }
    
    ~Manager() {
        cout << "  Manager destroyed" << endl;
    }
};

class Intern : public Employee {
private:
    string university;
    int duration;
    string mentor;
    
public:
    Intern(string n, double salary, string uni, int dur, string ment)
        : Employee(n, salary), university(uni), duration(dur), mentor(ment) {
        cout << "  Intern: from " << university << endl;
    }
    
    double calculateSalary() const override {
        return baseSalary * 0.5;
    }
    
    string getRole() const override {
        return "Intern";
    }
    
    void performDuties() const override {
        cout << name << " is learning from " << mentor << endl;
        cout << name << " is attending training sessions" << endl;
        cout << name << " is working on small tasks" << endl;
        cout << name << " is studying " << university << " materials" << endl;
    }
    
    void display() const override {
        Employee::display();
        cout << "University: " << university << endl;
        cout << "Duration: " << duration << " months" << endl;
        cout << "Mentor: " << mentor << endl;
    }
    
    ~Intern() {
        cout << "  Intern destroyed" << endl;
    }
};

int main() {
    vector<Employee*> employees;
    
    employees.push_back(new Developer("Alice Johnson", 60000, "C++", 8));
    employees.push_back(new Developer("Bob Smith", 65000, "Python", 6));
    employees.push_back(new Manager("Carol White", 80000, 5));
    employees.push_back(new Intern("David Brown", 30000, "MIT", 6, "Carol White"));
    
    // Add team members for manager
    Manager* mgr = dynamic_cast<Manager*>(employees[2]);
    if (mgr) {
        mgr->addTeamMember("Alice Johnson");
        mgr->addTeamMember("Bob Smith");
        mgr->addTeamMember("David Brown");
    }
    
    cout << "\n=== Employee Details ===" << endl;
    for (auto emp : employees) {
        emp->display();
        cout << endl;
    }
    
    cout << "=== Employee Duties ===" << endl;
    for (auto emp : employees) {
        cout << emp->getName() << " (" << emp->getRole() << "):" << endl;
        emp->performDuties();
        cout << endl;
    }
    
    // Cleanup
    for (auto emp : employees) {
        delete emp;
    }
    
    return 0;
}
```

### Example 2: Payment System with Abstract Base Class

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// Abstract base class
class Payment {
protected:
    double amount;
    string date;
    
public:
    Payment(double amt, string dt) : amount(amt), date(dt) {}
    
    // Pure virtual functions
    virtual void process() const = 0;
    virtual string getPaymentType() const = 0;
    virtual bool validate() const = 0;
    
    // Concrete methods
    virtual void display() const {
        cout << "Payment Type: " << getPaymentType() << endl;
        cout << "Amount: $" << amount << endl;
        cout << "Date: " << date << endl;
        cout << "Status: " << (validate() ? "Valid" : "Invalid") << endl;
    }
    
    virtual ~Payment() {}
};

class CreditCardPayment : public Payment {
private:
    string cardNumber;
    string expiryDate;
    string cvv;
    
public:
    CreditCardPayment(double amt, string dt, string card, string exp, string c)
        : Payment(amt, dt), cardNumber(card), expiryDate(exp), cvv(c) {}
    
    void process() const override {
        cout << "Processing credit card payment..." << endl;
        cout << "Card: ****" << cardNumber.substr(cardNumber.length() - 4) << endl;
        cout << "Amount: $" << amount << endl;
        cout << "Payment approved!" << endl;
    }
    
    string getPaymentType() const override {
        return "Credit Card";
    }
    
    bool validate() const override {
        return (cardNumber.length() == 16 && cvv.length() == 3);
    }
};

class PayPalPayment : public Payment {
private:
    string email;
    string password;
    
public:
    PayPalPayment(double amt, string dt, string em, string pass)
        : Payment(amt, dt), email(em), password(pass) {}
    
    void process() const override {
        cout << "Processing PayPal payment..." << endl;
        cout << "Account: " << email << endl;
        cout << "Amount: $" << amount << endl;
        cout << "Payment transferred!" << endl;
    }
    
    string getPaymentType() const override {
        return "PayPal";
    }
    
    bool validate() const override {
        return (email.find('@') != string::npos && password.length() >= 6);
    }
};

class BankTransferPayment : public Payment {
private:
    string accountNumber;
    string routingNumber;
    string bankName;
    
public:
    BankTransferPayment(double amt, string dt, string acc, string routing, string bank)
        : Payment(amt, dt), accountNumber(acc), routingNumber(routing), bankName(bank) {}
    
    void process() const override {
        cout << "Processing bank transfer..." << endl;
        cout << "Bank: " << bankName << endl;
        cout << "Account: ****" << accountNumber.substr(accountNumber.length() - 4) << endl;
        cout << "Amount: $" << amount << endl;
        cout << "Transfer initiated!" << endl;
    }
    
    string getPaymentType() const override {
        return "Bank Transfer";
    }
    
    bool validate() const override {
        return (accountNumber.length() >= 8 && routingNumber.length() == 9);
    }
};

int main() {
    vector<Payment*> payments;
    
    payments.push_back(new CreditCardPayment(150.00, "2024-01-15", "1234567812345678", "12/25", "123"));
    payments.push_back(new PayPalPayment(75.50, "2024-01-16", "user@example.com", "password123"));
    payments.push_back(new BankTransferPayment(500.00, "2024-01-17", "12345678", "987654321", "Chase Bank"));
    
    cout << "=== Payment Details ===" << endl;
    for (auto payment : payments) {
        payment->display();
        cout << endl;
    }
    
    cout << "=== Processing Payments ===" << endl;
    for (auto payment : payments) {
        payment->process();
        cout << endl;
    }
    
    for (auto payment : payments) {
        delete payment;
    }
    
    return 0;
}
```

## Abstract Class with Pure Virtual Destructor

```cpp
#include <iostream>
using namespace std;

class AbstractBase {
public:
    // Pure virtual destructor - must have implementation
    virtual ~AbstractBase() = 0;
};

// Pure virtual destructor must be defined
AbstractBase::~AbstractBase() {
    cout << "AbstractBase destructor" << endl;
}

class Derived : public AbstractBase {
public:
    ~Derived() {
        cout << "Derived destructor" << endl;
    }
};

int main() {
    // AbstractBase obj;  // Still cannot instantiate
    AbstractBase* ptr = new Derived();
    delete ptr;  // Calls Derived destructor, then AbstractBase destructor
    
    return 0;
}
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Trying to instantiate abstract class | Compilation error | Use derived classes |
| Forgetting to implement pure virtual functions | Derived class remains abstract | Implement all pure virtual functions |
| Missing pure virtual destructor definition | Linker error | Provide definition |
| Using abstract class as value type | Slicing | Use pointers/references |
| Not using override | Hidden functions | Use override keyword |

## Summary

- **Abstract classes** cannot be instantiated
- Contain at least one **pure virtual function**
- Define **interfaces** for derived classes
- Can contain **implemented methods** as well
- **Pure virtual destructor** must be defined
- Used for **polymorphic base classes**
- Enable **code reuse** through inheritance

## Basic Template

### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class AbstractClass {
public:
    virtual void pureVirtual() = 0;
    virtual void concreteMethod() {
        cout << "Concrete method" << endl;
    }
    virtual ~AbstractClass() {}
};

class ConcreteClass : public AbstractClass {
public:
    void pureVirtual() override {
        cout << "Implementation" << endl;
    }
};

int main() {
    // AbstractClass obj;  // Error!
    ConcreteClass obj;
    obj.pureVirtual();
    obj.concreteMethod();
    
    AbstractClass* ptr = new ConcreteClass();
    ptr->pureVirtual();
    delete ptr;
    
    return 0;
}
```
