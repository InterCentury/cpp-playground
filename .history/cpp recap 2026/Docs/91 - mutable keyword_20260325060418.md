# 91 - mutable Keyword in C++

## What is the mutable Keyword?
The `mutable` keyword allows a member variable of a const object to be modified even within const member functions. It's used for members that don't affect the externally visible state of the object, such as caches, counters, or debug information.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
#include <cmath>
using namespace std;

class Circle {
private:
    double radius;
    mutable double cachedArea;      // Cache for area calculation
    mutable bool areaValid;          // Flag if cache is valid
    mutable int areaCalculations;    // Counter for debugging
    
public:
    Circle(double r) : radius(r), cachedArea(0), areaValid(false), areaCalculations(0) {}
    
    // Const member function - can modify mutable members
    double getArea() const {
        areaCalculations++;  // mutable - OK
        if (!areaValid) {
            cachedArea = M_PI * radius * radius;  // mutable - OK
            areaValid = true;                      // mutable - OK
            cout << "Area calculated (cached)" << endl;
        } else {
            cout << "Area retrieved from cache" << endl;
        }
        return cachedArea;
    }
    
    void setRadius(double r) {
        radius = r;
        areaValid = false;  // Invalidate cache
    }
    
    int getCalculationCount() const { return areaCalculations; }
    
    void display() const {
        cout << "Radius: " << radius << ", Area: " << getArea() 
             << ", Calculations: " << areaCalculations << endl;
    }
};

int main() {
    const Circle c(5.0);  // const object
    
    c.display();  // First call - calculates
    c.display();  // Second call - uses cache
    
    // c.setRadius(10.0);  // Error! Cannot call non-const on const object
    
    cout << "\nTotal area calculations: " << c.getCalculationCount() << endl;
    
    return 0;
}
```

## Common Uses of mutable

### 1. Caching Results
```cpp
class ExpensiveComputation {
private:
    double input;
    mutable double cachedResult;
    mutable bool cacheValid;
    
public:
    ExpensiveComputation(double i) : input(i), cachedResult(0), cacheValid(false) {}
    
    double compute() const {
        if (!cacheValid) {
            // Expensive calculation
            cachedResult = input * input * input;  // Simulate expensive op
            cacheValid = true;
        }
        return cachedResult;
    }
};
```

### 2. Reference Counting / Usage Tracking
```cpp
class Resource {
private:
    int* data;
    mutable int refCount;
    
public:
    Resource() : data(new int[100]), refCount(0) {}
    
    void access() const {
        refCount++;  // Track access even on const objects
        // Use data...
    }
    
    int getRefCount() const { return refCount; }
    
    ~Resource() { delete[] data; }
};
```

### 3. Logging and Debugging
```cpp
class Logger {
private:
    string name;
    mutable int logCount;
    mutable time_t lastLogTime;
    
public:
    Logger(string n) : name(n), logCount(0), lastLogTime(0) {}
    
    void log(const string& message) const {
        logCount++;
        lastLogTime = time(nullptr);
        cout << "[" << name << "] " << message << endl;
    }
    
    int getLogCount() const { return logCount; }
    time_t getLastLogTime() const { return lastLogTime; }
};
```

### 4. Lazy Initialization
```cpp
class DatabaseConnection {
private:
    string connectionString;
    mutable bool connected;
    mutable void* connection;
    
public:
    DatabaseConnection(string conn) : connectionString(conn), connected(false), connection(nullptr) {}
    
    void* getConnection() const {
        if (!connected) {
            // Lazy initialization
            connection = connectToDatabase(connectionString);
            connected = true;
        }
        return connection;
    }
    
    void* connectToDatabase(const string& str) const {
        // Simulate connection
        cout << "Connecting to " << str << endl;
        return (void*)0x1234;
    }
};
```

## Practical Examples

### Example 1: Cache for Complex Calculations

#### Type 1: Single File
```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

class DataAnalyzer {
private:
    vector<double> data;
    mutable double cachedMean;
    mutable bool meanValid;
    mutable double cachedStdDev;
    mutable bool stdDevValid;
    mutable int computeCount;
    
    void computeMean() const {
        if (!meanValid) {
            double sum = 0;
            for (double d : data) sum += d;
            cachedMean = sum / data.size();
            meanValid = true;
            computeCount++;
            cout << "Mean computed: " << cachedMean << endl;
        }
    }
    
    void computeStdDev() const {
        if (!stdDevValid) {
            computeMean();  // Ensure mean is computed
            double sum = 0;
            for (double d : data) {
                double diff = d - cachedMean;
                sum += diff * diff;
            }
            cachedStdDev = sqrt(sum / data.size());
            stdDevValid = true;
            computeCount++;
            cout << "StdDev computed: " << cachedStdDev << endl;
        }
    }
    
public:
    DataAnalyzer(const vector<double>& d) : data(d), 
        cachedMean(0), meanValid(false), 
        cachedStdDev(0), stdDevValid(false), computeCount(0) {}
    
    double getMean() const {
        computeMean();
        return cachedMean;
    }
    
    double getStdDev() const {
        computeStdDev();
        return cachedStdDev;
    }
    
    void addData(double value) {
        data.push_back(value);
        meanValid = false;    // Invalidate cache
        stdDevValid = false;
    }
    
    int getComputeCount() const { return computeCount; }
    
    void display() const {
        cout << "Data size: " << data.size() << endl;
        cout << "Mean: " << getMean() << endl;
        cout << "StdDev: " << getStdDev() << endl;
        cout << "Computations: " << computeCount << endl;
    }
};

int main() {
    vector<double> data = {10, 20, 30, 40, 50};
    const DataAnalyzer analyzer(data);  // const object
    
    analyzer.display();  // Computes mean and stddev
    analyzer.display();  // Uses cached values
    
    // analyzer.addData(60);  // Error! Cannot modify const object
    
    cout << "\nTotal computations: " << analyzer.getComputeCount() << endl;
    
    return 0;
}
```

### Example 2: Thread-Safe Counter with mutable

#### Type 1: Single File
```cpp
#include <iostream>
#include <thread>
#include <mutex>
#include <vector>
using namespace std;

class ThreadSafeCounter {
private:
    mutable mutex mtx;
    mutable int count;
    
public:
    ThreadSafeCounter() : count(0) {}
    
    void increment() const {
        lock_guard<mutex> lock(mtx);  // mutable mutex - OK
        count++;                       // mutable int - OK
    }
    
    int getCount() const {
        lock_guard<mutex> lock(mtx);
        return count;
    }
};

void worker(const ThreadSafeCounter& counter, int iterations) {
    for (int i = 0; i < iterations; i++) {
        counter.increment();
    }
}

int main() {
    const ThreadSafeCounter counter;  // const object
    
    vector<thread> threads;
    for (int i = 0; i < 10; i++) {
        threads.emplace_back(worker, ref(counter), 1000);
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    cout << "Final count: " << counter.getCount() << endl;
    
    return 0;
}
```

### Example 3: Lazy Loading with mutable

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <fstream>
#include <sstream>
using namespace std;

class LazyFileReader {
private:
    string filename;
    mutable string content;
    mutable bool loaded;
    
    void loadContent() const {
        if (!loaded) {
            ifstream file(filename);
            if (file.is_open()) {
                stringstream buffer;
                buffer << file.rdbuf();
                content = buffer.str();
                loaded = true;
                cout << "File loaded: " << filename << " (" << content.length() << " bytes)" << endl;
            } else {
                content = "";
                cout << "Failed to load file: " << filename << endl;
            }
        }
    }
    
public:
    LazyFileReader(const string& fname) : filename(fname), loaded(false) {}
    
    string getContent() const {
        loadContent();
        return content;
    }
    
    int getSize() const {
        loadContent();
        return content.length();
    }
    
    void reload() {
        loaded = false;  // Force reload on next access
        cout << "Cache invalidated for: " << filename << endl;
    }
};

int main() {
    // Create a test file
    ofstream test("test.txt");
    test << "Hello World! This is test content.";
    test.close();
    
    const LazyFileReader reader("test.txt");
    
    cout << "First access - will load file:" << endl;
    cout << "Content: " << reader.getContent() << endl;
    
    cout << "\nSecond access - uses cache:" << endl;
    cout << "Size: " << reader.getSize() << " bytes" << endl;
    
    // reader.reload();  // Error! Cannot call non-const on const
    
    return 0;
}
```

## mutable vs Non-mutable Comparison

```cpp
#include <iostream>
using namespace std;

class WithoutMutable {
private:
    int value;
    int accessCount;  // Not mutable
    
public:
    WithoutMutable(int v) : value(v), accessCount(0) {}
    
    int getValue() {  // Can't be const
        accessCount++;
        return value;
    }
    
    int getAccessCount() const { return accessCount; }
};

class WithMutable {
private:
    int value;
    mutable int accessCount;  // Mutable
    
public:
    WithMutable(int v) : value(v), accessCount(0) {}
    
    int getValue() const {  // Can be const
        accessCount++;
        return value;
    }
    
    int getAccessCount() const { return accessCount; }
};

int main() {
    const WithoutMutable wom(10);
    // wom.getValue();  // Error! Can't call non-const
    
    const WithMutable wm(10);
    wm.getValue();  // OK - const function with mutable
    cout << "Access count: " << wm.getAccessCount() << endl;
    
    return 0;
}
```

## When to Use mutable

### ✅ Good Uses
```cpp
// Caching
mutable double cachedResult;
mutable bool cacheValid;

// Reference counting
mutable int refCount;

// Debugging/Logging
mutable int callCount;
mutable time_t lastAccess;

// Thread synchronization
mutable mutex mtx;

// Lazy initialization
mutable bool initialized;
mutable Resource* resource;
```

### ❌ Avoid
```cpp
// Never use mutable to bypass const correctness for logical state
mutable string name;  // Bad - name is logical state

// Don't use mutable for data that affects object identity
mutable int id;  // Bad - id is part of object identity

// Don't use mutable when you can redesign
mutable bool dirty;  // Consider redesign if possible
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Overusing mutable | Breaks const correctness | Use only for cache/debug |
| Modifying logical state | Inconsistent const semantics | Keep logical state const |
| Not invalidating cache | Stale results | Invalidate when source changes |
| Using mutable for everything | Loses const benefits | Minimize mutable usage |
| Thread safety with mutable | Race conditions | Use proper synchronization |

## Summary

- **mutable** allows modification in const member functions
- Used for **caching**, **logging**, **reference counting**, **lazy initialization**
- Does **not** affect logical constness of object
- Should be used **sparingly** and with clear intent
- Helps maintain **const correctness** while allowing necessary modifications
- Common in **thread-safe** classes with mutexes
- Essential for **performance optimizations** (caching)

## Basic Template

### Type 1: Single File
```cpp
#include <iostream>
#include <string>
using namespace std;

class CacheExample {
private:
    string data;
    mutable string cachedResult;
    mutable bool cacheValid;
    mutable int accessCount;
    
    string expensiveComputation() const {
        // Simulate expensive operation
        return data + "_processed";
    }
    
public:
    CacheExample(const string& d) : data(d), cacheValid(false), accessCount(0) {}
    
    string getResult() const {
        accessCount++;
        if (!cacheValid) {
            cachedResult = expensiveComputation();
            cacheValid = true;
            cout << "Computed new result" << endl;
        } else {
            cout << "Returned cached result" << endl;
        }
        return cachedResult;
    }
    
    void setData(const string& d) {
        data = d;
        cacheValid = false;  // Invalidate cache
    }
    
    int getAccessCount() const { return accessCount; }
};

int main() {
    const CacheExample cache("Hello");
    cache.getResult();  // Computes
    cache.getResult();  // Uses cache
    cout << "Accesses: " << cache.getAccessCount() << endl;
    return 0;
}
```
