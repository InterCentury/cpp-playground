# 43 - String Length in C++

## What is String Length?
String length refers to the number of characters in a string, excluding the null terminator (`\0`). In C++, the `string` class provides several methods to get the length of a string.

## Basic Length Methods

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text = "Hello World";
    
    // Different ways to get length
    cout << "String: \"" << text << "\"" << endl;
    cout << "length(): " << text.length() << endl;  // 11
    cout << "size(): " << text.size() << endl;      // 11 (same as length)
    cout << "sizeof(): " << sizeof(text) << " bytes (object size)" << endl;
    
    return 0;
}
```

## length() vs size() vs sizeof()

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str1 = "Hello";
    string str2 = "Hello World!";
    string str3 = "";
    
    cout << "String\t\tlength()\tsize()\tsizeof()" << endl;
    cout << "------\t\t--------\t------\t--------" << endl;
    
    cout << "\"Hello\"\t\t" << str1.length() << "\t\t" 
         << str1.size() << "\t\t" << sizeof(str1) << endl;
    
    cout << "\"Hello World!\"\t" << str2.length() << "\t\t" 
         << str2.size() << "\t\t" << sizeof(str2) << endl;
    
    cout << "Empty\t\t" << str3.length() << "\t\t" 
         << str3.size() << "\t\t" << sizeof(str3) << endl;
    
    // sizeof returns size of string object (usually constant)
    // length()/size() return actual character count
    
    return 0;
}
```

## Checking Empty Strings

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s1 = "";
    string s2 = " ";
    string s3 = "Hello";
    
    cout << boolalpha;
    
    // Using empty()
    cout << "s1 empty? " << s1.empty() << endl;      // true
    cout << "s2 empty? " << s2.empty() << endl;      // false (has space)
    cout << "s3 empty? " << s3.empty() << endl;      // false
    
    // Using length() to check empty
    cout << "\nChecking with length():" << endl;
    cout << "s1 length == 0? " << (s1.length() == 0) << endl;
    cout << "s2 length == 0? " << (s2.length() == 0) << endl;
    
    // Using size()
    cout << "\nChecking with size():" << endl;
    cout << "s1 size == 0? " << (s1.size() == 0) << endl;
    
    return 0;
}
```

## Practical Examples

### Example 1: Password Strength Checker
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string password;
    
    cout << "Enter password: ";
    cin >> password;
    
    int len = password.length();
    
    cout << "Password length: " << len << endl;
    
    if (len == 0) {
        cout << "Password cannot be empty!" << endl;
    }
    else if (len < 6) {
        cout << "Weak password (too short)" << endl;
    }
    else if (len < 10) {
        cout << "Medium strength password" << endl;
    }
    else {
        cout << "Strong password" << endl;
    }
    
    return 0;
}
```

### Example 2: Text Justification
```cpp
#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

int main() {
    string words[] = {"apple", "banana", "watermelon", "kiwi", "strawberry"};
    int size = sizeof(words) / sizeof(words[0]);
    
    // Find longest word
    int maxLen = 0;
    for (int i = 0; i < size; i++) {
        if (words[i].length() > maxLen) {
            maxLen = words[i].length();
        }
    }
    
    cout << "Words (right-aligned):" << endl;
    for (int i = 0; i < size; i++) {
        cout << setw(maxLen) << words[i] << " (length: " 
             << words[i].length() << ")" << endl;
    }
    
    return 0;
}
```

### Example 3: Truncate Long Text
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text = "This is a very long sentence that needs to be truncated.";
    int maxLength = 20;

    cout << text << endl;
    
    if (text.length() > maxLength) {
        string truncated = text.substr(0, maxLength) + "...";
        cout << "\nTruncated (" << truncated.length() << " chars):" << endl;
        cout << truncated << endl;
    }
    
    return 0;
}
```

### Example 4: Center Text
```cpp
#include <iostream>
#include <string>
using namespace std;

void printCentered(string text, int width) {
    int padding = width - text.length();
    int leftPad = padding / 2;
    int rightPad = padding - leftPad;
    
    for (int i = 0; i < leftPad; i++) cout << " ";
    cout << text;
    for (int i = 0; i < rightPad; i++) cout << " ";
}

int main() {
    string title = "WELCOME";
    int screenWidth = 50;
    
    cout << "+" << string(screenWidth, '-') << "+" << endl;
    cout << "|";
    printCentered(title, screenWidth - 2);
    cout << "|" << endl;
    cout << "+" << string(screenWidth, '-') << "+" << endl;
    
    return 0;
}
```

### Example 5: Progress Indicator
```cpp
#include <iostream>
#include <string>
#include <thread>
#include <chrono>
using namespace std;

int main() {
    string status = "Loading...";
    string dots;
    
    for (int i = 0; i <= 10; i++) {
        // Clear line
        cout << "\r" << status;
        
        // Add dots based on progress
        dots = string(i % 4, '.');  // Cycles 0-3 dots
        cout << dots;
        
        // Pad with spaces to clear previous
        cout << string(10 - dots.length(), ' ');
        
        cout.flush();
        this_thread::sleep_for(chrono::milliseconds(300));

    return 0;
}
```

## Length with Different Character Types

### ASCII vs Unicode (Multi-byte characters)
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    // ASCII characters (1 byte each)
    string ascii = "Hello";
    cout << "ASCII string: " << ascii << endl;
    cout << "Length: " << ascii.length() << endl;
    cout << "Bytes: " << sizeof(char) * ascii.length() << endl << endl;
    
    // UTF-8 with multi-byte characters
    string utf8 = "café";  // 'é' is 2 bytes in UTF-8
    cout << "UTF-8 string: " << utf8 << endl;
    cout << "length() reports: " << utf8.length() << " (bytes, not characters!)" << endl;
    cout << "Actual characters: 4" << endl;
    
    return 0;
}
```

## String Length with User Input

```cpp
#include <iostream>
#include <string>
#include <limits>
using namespace std;

int main() {
    string input;
    
    cout << "Enter a line of text: ";
    getline(cin, input);
    
    cout << "You entered: \"" << input << "\"" << endl;
    cout << "Length: " << input.length() << " characters" << endl;
    
    // Character-by-character analysis
    int letters = 0, digits = 0, spaces = 0, others = 0;
    
    for (char c : input) {
        if (isalpha(c)) letters++;
        else if (isdigit(c)) digits++;
        else if (isspace(c)) spaces++;
        else others++;
    }
    
    cout << "\nBreakdown:" << endl;
    cout << "  Letters: " << letters << endl;
    cout << "  Digits: " << digits << endl;
    cout << "  Spaces: " << spaces << endl;
    cout << "  Other: " << others << endl;
    
    return 0;
}
```

## C-style String Length

```cpp
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    // C-style strings
    char str1[] = "Hello";
    char str2[20] = "World";
    const char* str3 = "C++";
    
    // Using strlen() from <cstring>
    cout << "C-style string lengths:" << endl;
    cout << "str1: " << strlen(str1) << endl;
    cout << "str2: " << strlen(str2) << endl;
    cout << "str3: " << strlen(str3) << endl;
    
    // Manual length calculation
    int len = 0;
    while (str1[len] != '\0') {
        len++;
    }
    cout << "Manual length of str1: " << len << endl;
    
    // sizeof vs strlen
    cout << "\nsizeof(str1): " << sizeof(str1) << " (includes null terminator)" << endl;
    cout << "strlen(str1): " << strlen(str1) << " (excludes null terminator)" << endl;
    
    return 0;
}
```

## Common Mistakes

### Mistake 1: Confusing length() with sizeof()
```cpp
string str = "Hello";

// Wrong - sizeof gives object size, not string length
int len = sizeof(str);  // Usually 24-32 bytes (object overhead)

// Correct
int len = str.length();  // 5
```

### Mistake 2: Off-by-One Errors
```cpp
string str = "Hello";

// Wrong - last index is length-1
for (int i = 0; i <= str.length(); i++) {  // Goes one past end
    cout << str[i];  // Prints garbage at end
}

// Correct
for (int i = 0; i < str.length(); i++) {
    cout << str[i];
}
```

### Mistake 3: Assuming length() is O(1) (It is, but...)
```cpp
// This is fine - length() is constant time
for (size_t i = 0; i < str.length(); i++) {
    // Do something
}

// But avoid calling it repeatedly if it changes
for (size_t i = 0; i < str.length(); i++) {
    str += "x";  // length changes each iteration!
    // Now you have a different condition each time
}
```

### Mistake 4: Not Checking for Empty String
```cpp
string str;
cout << "First character: " << str[0] << endl;  // Undefined behavior!

// Correct
if (!str.empty()) {
    cout << "First character: " << str[0] << endl;
}
```

### Mistake 5: Using strlen on C++ strings
```cpp
string str = "Hello";

// Wrong - strlen expects char*
int len = strlen(str);  // Compiler error!

// Correct
int len = str.length();

// If you really need C-string
int len = strlen(str.c_str());  // Works but unnecessary
```

## Quick Reference

| Method | Returns | Example |
|--------|---------|---------|
| `length()` | Number of characters | `str.length()` |
| `size()` | Same as length() | `str.size()` |
| `empty()` | true if length == 0 | `str.empty()` |
| `sizeof()` | Size of string object (not content) | `sizeof(str)` |
| `strlen()` | Length of C-string | `strlen(cstr)` |

## Summary

- **length()** and **size()** are identical - both return character count
- **empty()** checks if string has no characters
- **sizeof()** returns object size (not string content length)
- Always check bounds before accessing characters
- Length excludes null terminator (unlike C-style strings)
- length() is O(1) constant time operation
- Use length() for loops and conditions
- Empty string has length 0
- Multi-byte characters (UTF-8) may cause length() to return bytes, not characters
- C-strings use **strlen()** from `<cstring>`

## Basic Template
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text;
    
    cout << "Enter some text: ";
    getline(cin, text);
    
    // Basic length info
    cout << "\n--- String Information ---" << endl;
    cout << "Text: \"" << text << "\"" << endl;
    cout << "Length: " << text.length() << " characters" << endl;
    cout << "Empty? " << (text.empty() ? "Yes" : "No") << endl;
    
    // Character access
    if (!text.empty()) {
        cout << "First character: " << text[0] << endl;
        cout << "Last character: " << text[text.length() - 1] << endl;
    }
    
    // Loop using length
    cout << "\nCharacters: ";
    for (size_t i = 0; i < text.length(); i++) {
        cout << text[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
