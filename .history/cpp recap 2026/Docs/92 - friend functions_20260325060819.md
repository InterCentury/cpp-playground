# 92 - Friend Functions in C++

## What are Friend Functions?
A friend function is a non-member function that has access to the private and protected members of a class. It is declared using the `friend` keyword inside the class. Friend functions are not member functions but can access private data.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
using namespace std;

class Complex {
private:
    double real;
    double imag;
    
public:
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}
    
    // Friend function declaration
    friend Complex add(const Complex& a, const Complex& b);
    friend void display(const Complex& c);
    friend Complex operator+(const Complex& a, const Complex& b);
};

// Friend function definitions - can access private members
Complex add(const Complex& a, const Complex& b) {
    return Complex(a.real + b.real, a.imag + b.imag);
}

void display(const Complex& c) {
    cout << c.real << " + " << c.imag << "i" << endl;
}

Complex operator+(const Complex& a, const Complex& b) {
    return Complex(a.real + b.real, a.imag + b.imag);
}

int main() {
    Complex c1(3, 4);
    Complex c2(1, 2);
    
    cout << "c1: "; display(c1);
    cout << "c2: "; display(c2);
    
    Complex sum = add(c1, c2);
    cout << "Sum: "; display(sum);
    
    Complex sum2 = c1 + c2;
    cout << "Sum using operator: "; display(sum2);
    
    return 0;
}
```

## Friend Functions vs Member Functions

```cpp
#include <iostream>
using namespace std;

class Rectangle {
private:
    double length;
    double width;
    
public:
    Rectangle(double l, double w) : length(l), width(w) {}
    
    // Member function
    double area() const { return length * width; }
    
    // Friend function declaration
    friend double perimeter(const Rectangle& r);
};

// Friend function - can access private members
double perimeter(const Rectangle& r) {
    return 2 * (r.length + r.width);  // Direct access
}

int main() {
    Rectangle rect(5, 3);
    
    cout << "Area (member): " << rect.area() << endl;
    cout << "Perimeter (friend): " << perimeter(rect) << endl;
    
    return 0;
}
```

## Practical Examples

### Example 1: Fraction Class with Friend Functions

#### Type 1: Single File
```cpp
#include <iostream>
#include <cmath>
using namespace std;

class Fraction {
private:
    int numerator;
    int denominator;
    
    void simplify() {
        int gcd = abs(gcdValue(numerator, denominator));
        numerator /= gcd;
        denominator /= gcd;
        if (denominator < 0) {
            numerator = -numerator;
            denominator = -denominator;
        }
    }
    
    int gcdValue(int a, int b) const {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
    
public:
    Fraction(int num = 0, int den = 1) : numerator(num), denominator(den) {
        if (denominator == 0) denominator = 1;
        simplify();
    }
    
    // Friend functions
    friend Fraction operator+(const Fraction& a, const Fraction& b);
    friend Fraction operator-(const Fraction& a, const Fraction& b);
    friend Fraction operator*(const Fraction& a, const Fraction& b);
    friend Fraction operator/(const Fraction& a, const Fraction& b);
    friend ostream& operator<<(ostream& os, const Fraction& f);
    friend bool operator==(const Fraction& a, const Fraction& b);
};

Fraction operator+(const Fraction& a, const Fraction& b) {
    return Fraction(a.numerator * b.denominator + b.numerator * a.denominator,
                    a.denominator * b.denominator);
}

Fraction operator-(const Fraction& a, const Fraction& b) {
    return Fraction(a.numerator * b.denominator - b.numerator * a.denominator,
                    a.denominator * b.denominator);
}

Fraction operator*(const Fraction& a, const Fraction& b) {
    return Fraction(a.numerator * b.numerator, a.denominator * b.denominator);
}

Fraction operator/(const Fraction& a, const Fraction& b) {
    return Fraction(a.numerator * b.denominator, a.denominator * b.numerator);
}

ostream& operator<<(ostream& os, const Fraction& f) {
    if (f.denominator == 1) {
        os << f.numerator;
    } else {
        os << f.numerator << "/" << f.denominator;
    }
    return os;
}

bool operator==(const Fraction& a, const Fraction& b) {
    return a.numerator * b.denominator == b.numerator * a.denominator;
}

int main() {
    Fraction f1(1, 2);
    Fraction f2(2, 3);
    
    cout << "f1 = " << f1 << endl;
    cout << "f2 = " << f2 << endl;
    cout << "f1 + f2 = " << f1 + f2 << endl;
    cout << "f1 - f2 = " << f1 - f2 << endl;
    cout << "f1 * f2 = " << f1 * f2 << endl;
    cout << "f1 / f2 = " << f1 / f2 << endl;
    cout << "f1 == f2? " << (f1 == f2 ? "Yes" : "No") << endl;
    
    return 0;
}
```

### Example 2: Matrix Class with Friend Functions

#### Type 1: Single File
```cpp
#include <iostream>
#include <vector>
using namespace std;

class Matrix {
private:
    int rows;
    int cols;
    vector<vector<int>> data;
    
public:
    Matrix(int r, int c) : rows(r), cols(c), data(r, vector<int>(c, 0)) {}
    
    int& at(int r, int c) { return data[r][c]; }
    const int& at(int r, int c) const { return data[r][c]; }
    
    // Friend functions
    friend Matrix operator+(const Matrix& a, const Matrix& b);
    friend Matrix operator*(const Matrix& a, const Matrix& b);
    friend ostream& operator<<(ostream& os, const Matrix& m);
    friend bool operator==(const Matrix& a, const Matrix& b);
};

Matrix operator+(const Matrix& a, const Matrix& b) {
    if (a.rows != b.rows || a.cols != b.cols) {
        cout << "Matrix dimensions mismatch!" << endl;
        return Matrix(0, 0);
    }
    
    Matrix result(a.rows, a.cols);
    for (int i = 0; i < a.rows; i++) {
        for (int j = 0; j < a.cols; j++) {
            result.data[i][j] = a.data[i][j] + b.data[i][j];
        }
    }
    return result;
}

Matrix operator*(const Matrix& a, const Matrix& b) {
    if (a.cols != b.rows) {
        cout << "Matrix dimensions incompatible for multiplication!" << endl;
        return Matrix(0, 0);
    }
    
    Matrix result(a.rows, b.cols);
    for (int i = 0; i < a.rows; i++) {
        for (int j = 0; j < b.cols; j++) {
            int sum = 0;
            for (int k = 0; k < a.cols; k++) {
                sum += a.data[i][k] * b.data[k][j];
            }
            result.data[i][j] = sum;
        }
    }
    return result;
}

ostream& operator<<(ostream& os, const Matrix& m) {
    for (int i = 0; i < m.rows; i++) {
        for (int j = 0; j < m.cols; j++) {
            os << m.data[i][j] << "\t";
        }
        os << endl;
    }
    return os;
}

bool operator==(const Matrix& a, const Matrix& b) {
    if (a.rows != b.rows || a.cols != b.cols) return false;
    
    for (int i = 0; i < a.rows; i++) {
        for (int j = 0; j < a.cols; j++) {
            if (a.data[i][j] != b.data[i][j]) return false;
        }
    }
    return true;
}

int main() {
    Matrix m1(2, 2);
    Matrix m2(2, 2);
    
    m1.at(0,0) = 1; m1.at(0,1) = 2;
    m1.at(1,0) = 3; m1.at(1,1) = 4;
    
    m2.at(0,0) = 5; m2.at(0,1) = 6;
    m2.at(1,0) = 7; m2.at(1,1) = 8;
    
    cout << "Matrix 1:" << endl << m1;
    cout << "Matrix 2:" << endl << m2;
    cout << "Sum:" << endl << m1 + m2;
    cout << "Product:" << endl << m1 * m2;
    
    return 0;
}
```

### Example 3: Box Class with Comparison Friends

#### Type 1: Single File
```cpp
#include <iostream>
#include <cmath>
using namespace std;

class Box {
private:
    double length;
    double width;
    double height;
    
public:
    Box(double l, double w, double h) : length(l), width(w), height(h) {}
    
    double volume() const { return length * width * height; }
    
    // Friend functions for comparison
    friend bool operator<(const Box& a, const Box& b);
    friend bool operator>(const Box& a, const Box& b);
    friend bool operator==(const Box& a, const Box& b);
    friend ostream& operator<<(ostream& os, const Box& b);
};

bool operator<(const Box& a, const Box& b) {
    return a.volume() < b.volume();
}

bool operator>(const Box& a, const Box& b) {
    return a.volume() > b.volume();
}

bool operator==(const Box& a, const Box& b) {
    return a.length == b.length && a.width == b.width && a.height == b.height;
}

ostream& operator<<(ostream& os, const Box& b) {
    os << b.length << " x " << b.width << " x " << b.height;
    return os;
}

int main() {
    Box b1(5, 3, 2);
    Box b2(4, 4, 2);
    Box b3(5, 3, 2);
    
    cout << "Box 1: " << b1 << " (Volume: " << b1.volume() << ")" << endl;
    cout << "Box 2: " << b2 << " (Volume: " << b2.volume() << ")" << endl;
    cout << "Box 3: " << b3 << " (Volume: " << b3.volume() << ")" << endl;
    
    cout << "b1 < b2: " << (b1 < b2 ? "Yes" : "No") << endl;
    cout << "b1 > b2: " << (b1 > b2 ? "Yes" : "No") << endl;
    cout << "b1 == b3: " << (b1 == b3 ? "Yes" : "No") << endl;
    
    return 0;
}
```

## Friend Function vs Friend Class

```cpp
#include <iostream>
#include <string>
using namespace std;

class Account {
private:
    double balance;
    string owner;
    
public:
    Account(string o, double b) : owner(o), balance(b) {}
    
    // Friend class
    friend class AccountManager;
    
    // Friend function
    friend void displayBalance(const Account& acc);
};

class AccountManager {
public:
    void deposit(Account& acc, double amount) {
        acc.balance += amount;  // Can access private
        cout << "Deposited to " << acc.owner << endl;
    }
    
    void withdraw(Account& acc, double amount) {
        if (amount <= acc.balance) {
            acc.balance -= amount;
            cout << "Withdrawn from " << acc.owner << endl;
        }
    }
};

void displayBalance(const Account& acc) {
    cout << acc.owner << " balance: $" << acc.balance << endl;
}

int main() {
    Account acc("John", 1000);
    AccountManager mgr;
    
    displayBalance(acc);
    mgr.deposit(acc, 500);
    displayBalance(acc);
    mgr.withdraw(acc, 200);
    displayBalance(acc);
    
    return 0;
}
```

## Friend Functions Across Multiple Classes

```cpp
#include <iostream>
#include <string>
using namespace std;

class Vector;
class Point;

class Point {
private:
    double x, y;
    
public:
    Point(double x = 0, double y = 0) : x(x), y(y) {}
    
    // Friend function
    friend Vector operator-(const Point& a, const Point& b);
    friend ostream& operator<<(ostream& os, const Point& p);
};

class Vector {
private:
    double dx, dy;
    
public:
    Vector(double dx = 0, double dy = 0) : dx(dx), dy(dy) {}
    
    double magnitude() const {
        return sqrt(dx*dx + dy*dy);
    }
    
    friend Vector operator-(const Point& a, const Point& b);
    friend ostream& operator<<(ostream& os, const Vector& v);
};

Vector operator-(const Point& a, const Point& b) {
    return Vector(a.x - b.x, a.y - b.y);
}

ostream& operator<<(ostream& os, const Point& p) {
    os << "(" << p.x << ", " << p.y << ")";
    return os;
}

ostream& operator<<(ostream& os, const Vector& v) {
    os << "<" << v.dx << ", " << v.dy << ">";
    return os;
}

int main() {
    Point p1(3, 4);
    Point p2(1, 2);
    
    Vector v = p1 - p2;
    
    cout << "p1: " << p1 << endl;
    cout << "p2: " << p2 << endl;
    cout << "Vector: " << v << endl;
    cout << "Magnitude: " << v.magnitude() << endl;
    
    return 0;
}
```

## Friend Function vs Static Member Function

```cpp
#include <iostream>
#include <string>
using namespace std;

class Config {
private:
    static string server;
    static int port;
    
public:
    static void setServer(const string& s) { server = s; }
    static void setPort(int p) { port = p; }
    
    // Friend function
    friend void printConfig();
    
    // Static member function
    static void printConfigStatic() {
        cout << "Server: " << server << ", Port: " << port << endl;
    }
};

string Config::server = "localhost";
int Config::port = 8080;

void printConfig() {
    cout << "Server (friend): " << Config::server << ", Port: " << Config::port << endl;
}

int main() {
    Config::printConfigStatic();
    printConfig();
    
    return 0;
}
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Friend declaration in wrong place | Syntax error | Declare inside class |
| Forgetting function definition | Linker error | Define friend function |
| Overusing friendship | Breaks encapsulation | Use only when necessary |
| Friend function not declared before use | Compilation error | Declare prototype |
| Friend function accessing non-existent members | Runtime error | Ensure members exist |

## When to Use Friend Functions

### ✅ Good Uses
```cpp
// Operator overloading (especially binary operators)
friend Complex operator+(const Complex& a, const Complex& b);

// Stream operators
friend ostream& operator<<(ostream& os, const MyClass& obj);
friend istream& operator>>(istream& is, MyClass& obj);

// Functions that need access to multiple classes
friend Vector operator-(const Point& a, const Point& b);

// Unit testing
friend class TestClass;
```

### ❌ Avoid
```cpp
// When member function works fine
// Instead of: friend void setValue(MyClass& obj, int val);
// Use: void setValue(int val);

// When breaking encapsulation unnecessarily
// Instead of: friend void printAllDetails(MyClass& obj);
// Use: obj.printAllDetails();

// For every function that accesses private members
// Only when absolutely necessary
```

## Summary

- **Friend functions** can access private/protected members
- Declared inside class with `friend` keyword
- Not member functions (no `this` pointer)
- Can be defined outside class
- Useful for **operator overloading**
- Can be **friend functions** or **friend classes**
- **Breaks encapsulation** - use sparingly
- Common in **stream operators** and **binary operators**

## Basic Template

### Type 1: Single File
```cpp
#include <iostream>
using namespace std;

class MyClass {
private:
    int data;
    
public:
    MyClass(int d) : data(d) {}
    
    // Friend function declaration
    friend void showData(const MyClass& obj);
    friend MyClass add(const MyClass& a, const MyClass& b);
```
