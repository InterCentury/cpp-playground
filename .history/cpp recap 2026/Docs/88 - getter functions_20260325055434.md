# 88 - Getter Functions in C++

## What are Getter Functions?
Getter functions (also called accessors) are public member functions that provide read-only access to private member variables. They encapsulate the data and allow controlled access while maintaining data hiding.

## Type 1: Single File Implementation

```cpp
#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    string name;
    int rollNumber;
    double gpa;
    string email;
    
public:
    // Constructor
    Student(string n, int roll, double g, string e) 
        : name(n), rollNumber(roll), gpa(g), email(e) {}
    
    // Getter functions
    string getName() const { return name; }
    int getRollNumber() const { return rollNumber; }
    double getGpa() const { return gpa; }
    string getEmail() const { return email; }
    
    // Computed getter
    string getGrade() const {
        if (gpa >= 3.5) return "A";
        if (gpa >= 3.0) return "B";
        if (gpa >= 2.5) return "C";
        if (gpa >= 2.0) return "D";
        return "F";
    }
    
    void display() const {
        cout << "Name: " << getName() << endl;
        cout << "Roll: " << getRollNumber() << endl;
        cout << "GPA: " << getGpa() << endl;
        cout << "Grade: " << getGrade() << endl;
        cout << "Email: " << getEmail() << endl;
    }
};

int main() {
    Student s("John Doe", 101, 3.7, "john@example.com");
    s.display();
    
    // Read-only access
    cout << "\nAccess via getters: " << s.getName() << " has GPA " << s.getGpa();
    
    return 0;
}
```

## Types of Getter Functions

### 1. Basic Getter (Return by Value)
```cpp
class Person {
private:
    string name;
    int age;
    
public:
    // Returns a copy - safe but may be expensive
    string getName() const { return name; }
    int getAge() const { return age; }
};
```

### 2. Getter Returning const Reference (Efficient)
```cpp
class LargeObject {
private:
    string description;
    vector<int> data;
    
public:
    // Returns const reference - no copy, read-only
    const string& getDescription() const { return description; }
    const vector<int>& getData() const { return data; }
};
```

### 3. Getter Returning by Pointer
```cpp
class DataBuffer {
private:
    int* buffer;
    int size;
    
public:
    // Returns pointer - can be null
    const int* getBuffer() const { return buffer; }
    int getSize() const { return size; }
};
```

### 4. Computed Getter
```cpp
class Rectangle {
private:
    double length;
    double width;
    
public:
    Rectangle(double l, double w) : length(l), width(w) {}
    
    // Computed values - no stored member
    double getArea() const { return length * width; }
    double getPerimeter() const { return 2 * (length + width); }
    double getDiagonal() const { return sqrt(length*length + width*width); }
};
```

### 5. Getter with Logic/Validation
```cpp
class BankAccount {
private:
    string accountNumber;
    double balance;
    string pin;
    
public:
    // Getter with validation
    double getBalance(const string& inputPin) const {
        if (inputPin == pin) {
            return balance;
        }
        return -1;  // Error indicator
    }
    
    // Masked getter
    string getAccountNumber() const {
        return "****" + accountNumber.substr(accountNumber.length() - 4);
    }
};
```

## Practical Examples

### Example 1: Employee Class with Various Getters

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

class Employee {
private:
    int id;
    string name;
    double salary;
    string department;
    vector<string> skills;
    static int nextId;
    
    string capitalize(string str) const {
        if (!str.empty()) {
            str[0] = toupper(str[0]);
        }
        return str;
    }
    
public:
    Employee(string n, double s, string dept) 
        : id(nextId++), name(n), salary(s), department(dept) {}
    
    // Basic getters
    int getId() const { return id; }
    string getName() const { return capitalize(name); }
    double getSalary() const { return salary; }
    string getDepartment() const { return capitalize(department); }
    
    // Getter with formatting
    string getFormattedSalary() const {
        char buffer[50];
        sprintf(buffer, "$%.2f", salary);
        return string(buffer);
    }
    
    // Getter with calculation
    double getAnnualSalary() const { return salary * 12; }
    double getBonus() const { return salary * 0.1; }
    
    // Getter returning const reference (efficient)
    const vector<string>& getSkills() const { return skills; }
    
    // Getter with logic
    string getSalaryGrade() const {
        if (salary >= 10000) return "Executive";
        if (salary >= 7000) return "Senior";
        if (salary >= 4000) return "Mid-Level";
        return "Junior";
    }
    
    // Add skill (mutator)
    void addSkill(string skill) {
        skills.push_back(skill);
    }
    
    void display() const {
        cout << "ID: " << getId() << endl;
        cout << "Name: " << getName() << endl;
        cout << "Salary: " << getFormattedSalary() << endl;
        cout << "Annual: " << getAnnualSalary() << endl;
        cout << "Grade: " << getSalaryGrade() << endl;
        cout << "Department: " << getDepartment() << endl;
        cout << "Skills: ";
        for (const auto& s : getSkills()) {
            cout << s << " ";
        }
        cout << endl;
    }
};

int Employee::nextId = 1000;

int main() {
    Employee e1("john doe", 5000, "engineering");
    e1.addSkill("C++");
    e1.addSkill("Python");
    e1.addSkill("SQL");
    
    e1.display();
    
    cout << "\nAccess via getters: " << e1.getName() 
         << " earns " << e1.getFormattedSalary() << endl;
    
    return 0;
}
```

### Example 2: Circle Class with Computed Getters

#### Type 1: Single File
```cpp
#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

class Circle {
private:
    double radius;
    const double PI = 3.14159265358979;
    
public:
    Circle(double r) : radius(r) {}
    
    // Basic getter
    double getRadius() const { return radius; }
    
    // Computed getters
    double getDiameter() const { return 2 * radius; }
    double getCircumference() const { return 2 * PI * radius; }
    double getArea() const { return PI * radius * radius; }
    
    // Formatted getters
    string getFormattedArea() const {
        char buffer[50];
        sprintf(buffer, "%.2f", getArea());
        return string(buffer);
    }
    
    string getFormattedCircumference() const {
        char buffer[50];
        sprintf(buffer, "%.2f", getCircumference());
        return string(buffer);
    }
    
    // Getter with logic
    string getSizeCategory() const {
        if (radius < 1) return "Tiny";
        if (radius < 5) return "Small";
        if (radius < 10) return "Medium";
        if (radius < 20) return "Large";
        return "Huge";
    }
    
    void display() const {
        cout << fixed << setprecision(2);
        cout << "Circle (r=" << radius << ")" << endl;
        cout << "  Diameter: " << getDiameter() << endl;
        cout << "  Circumference: " << getCircumference() << endl;
        cout << "  Area: " << getArea() << endl;
        cout << "  Category: " << getSizeCategory() << endl;
    }
};

int main() {
    Circle c1(5.0);
    Circle c2(0.5);
    Circle c3(15.0);
    
    c1.display();
    cout << endl;
    c2.display();
    cout << endl;
    c3.display();
    
    cout << "\nIndividual getters: " << c1.getFormattedArea() << endl;
    
    return 0;
}
```

### Example 3: Bank Account with Security Getters

#### Type 1: Single File
```cpp
#include <iostream>
#include <string>
#include <ctime>
using namespace std;

class BankAccount {
private:
    string accountNumber;
    string accountHolder;
    double balance;
    string pin;
    time_t lastAccess;
    int failedAttempts;
    
    bool authenticate(string inputPin) const {
        return pin == inputPin;
    }
    
    void updateAccessTime() {
        lastAccess = time(nullptr);
    }
    
public:
    BankAccount(string number, string holder, double initial, string p)
        : accountNumber(number), accountHolder(holder), 
          balance(initial), pin(p), failedAttempts(0) {
        updateAccessTime();
    }
    
    // Secure getter with authentication
    double getBalance(string inputPin) {
        if (authenticate(inputPin)) {
            updateAccessTime();
            failedAttempts = 0;
            return balance;
        }
        failedAttempts++;
        return -1;
    }
    
    // Masked getter
    string getAccountNumber() const {
        if (accountNumber.length() > 8) {
            return "****" + accountNumber.substr(accountNumber.length() - 4);
        }
        return "****";
    }
    
    string getAccountHolder() const {
        return accountHolder;
    }
    
    // Getter with status
    string getLastAccess() const {
        return ctime(&lastAccess);
    }
    
    int getFailedAttempts() const {
        return failedAttempts;
    }
    
    bool isLocked() const {
        return failedAttempts >= 3;
    }
    
    // Getter with formatting
    string getFormattedBalance() const {
        char buffer[50];
        sprintf(buffer, "$%.2f", balance);
        return string(buffer);
    }
};

int main() {
    BankAccount acc("1234567890", "John Doe", 1000.00, "1234");
    
    cout << "Account: " << acc.getAccountNumber() << endl;
    cout << "Holder: " << acc.getAccountHolder() << endl;
    
    double balance = acc.getBalance("1234");
    if (balance >= 0) {
        cout << "Balance: " << acc.getFormattedBalance() << endl;
    } else {
        cout << "Wrong PIN!" << endl;
    }
    
    // Wrong PIN attempt
    balance = acc.getBalance("9999");
    cout << "Failed attempts: " << acc.getFailedAttempts() << endl;
    
    return 0;
}
```

## Getter Return Types - Best Practices

```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

class BestPractice {
private:
    int primitive;           // Small type
    string largeString;      // Large type
    vector<int> largeVector; // Large container
    int* array;              // Pointer
    int size;
    
public:
    // For primitive types - return by value
    int getPrimitive() const { return primitive; }
    
    // For large objects - return const reference
    const string& getLargeString() const { return largeString; }
    const vector<int>& getLargeVector() const { return largeVector; }
    
    // For pointers - return const pointer
    const int* getArray() const { return array; }
    int getSize() const { return size; }
    
    // For optional values - return by value or pointer
    int* getData() { return array; }  // Non-const for modification
    const int* getData() const { return array; }  // Const version
};
```

## Common Mistakes

| Mistake | Problem | Solution |
|---------|---------|----------|
| Returning non-const reference | Breaks encapsulation | Return const reference or by value |
| Missing const keyword | Can't call on const objects | Add const to getters |
| Returning pointer to internal data | Can be modified externally | Return const pointer |
| Expensive copies | Performance hit | Return const reference for large objects |
| Not handling edge cases | Garbage values | Add validation in getters |

## Summary

- **Getters** provide controlled read-only access to private members
- **const correctness**: always mark getters as `const`
- **Return by value** for small/primitive types
- **Return const reference** for large objects (efficiency)
- **Computed getters** calculate values on demand
- **Security getters** can include authentication/validation
- **Encapsulation** keeps internal representation hidden

## Basic Template

### Type 1: Single File
```cpp
#include <iostream>
#include <string>
using namespace std;

class MyClass {
private:
    int id;
    string name;
    double value;
    
public:
    MyClass(int i, string n, double v) : id(i), name(n), value(v) {}
    
    // Basic getters (const)
    int getId() const { return id; }
    string getName() const { return name; }
    double getValue() const { return value; }
    
    // Computed getter
    string getFormattedValue() const {
        char buffer[50];
        sprintf(buffer, "%.2f", value);
        return string(buffer);
    }
    
    void display() const {
        cout << "ID: " << getId() << endl;
        cout << "Name: " << getName() << endl;
        cout << "Value: " << getFormattedValue() << endl;
    }
};

int main() {
    MyClass obj(101, "Test", 42.5);
    obj.display();
    cout << "\nName via getter: " << obj.getName() << endl;
    return 0;
}
```
