# 50 - return Statement in C++

## What is the return Statement?
The `return` statement ends the execution of a function and returns control to the calling function. If the function has a non-void return type, it also returns a value.

## Basic Syntax

```cpp
return;        // For void functions
return value;  // For functions that return a value
```

## Simple Examples

```cpp
#include <iostream> 
using namespace std;

// void function - no return value needed
void sayHello() {
    cout << "Hello!" << endl;
    return;  // Optional in void functions
}

// Function that returns a value
int add(int a, int b) {
    return a + b;  // Returns the sum
}

int main() {
    sayHello();
    
    int result = add(5, 3);
    cout << "Sum: " << result << endl;
    
    return 0;  // Returns 0 to operating system
}
```

### Returning reference
```cpp
#include <iostream>
using namespace std;

int& getMax(int &a, int &b) {
    if (a > b) {
        return a;  // Returns reference to a
    } else {
        return b;  // Returns reference to b
    }
}

int main() {
    int x = 10, y = 20;
    
    cout << "Before: x=" << x << ", y=" << y << endl;
    
    getMax(x, y) = 100;  // Assigns 100 to the larger variable
    
    cout << "After: x=" << x << ", y=" << y << endl;  // y becomes 100
    
    return 0;
}
```

## Early Return

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool searchValue(const vector<int>& vec, int target) {
    for (int val : vec) {
        if (val == target) {
            return true;  // Early return when found
        }
    }
    return false;  // Return after loop if not found
}

int findFirstEven(const int arr[], int size) {
    for (int i = 0; i < size; i++) {
        if (arr[i] % 2 == 0) {
            return i;  // Return immediately when found
        }
    }
    return -1;  // Return -1 if no even number
}

void processNumbers(int num) {
    if (num < 0) {
        cout << "Negative number - stopping" << endl;
        return;  // Early exit
    }
    
    // Process positive numbers
    cout << "Processing: " << num << endl;
    // More processing...
}

int main() {
    vector<int> numbers = {1, 3, 5, 7, 8, 9};
    cout << "Found 8? " << boolalpha << searchValue(numbers, 8) << endl;
    
    int arr[] = {1, 3, 5, 7, 9, 2, 4};
    int pos = findFirstEven(arr, 7);
    if (pos != -1) {
        cout << "First even at index: " << pos << endl;
    }
    
    processNumbers(-5);
    processNumbers(10);
    
    return 0;
}
```

## return in Recursive Functions

```cpp
#include <iostream>
using namespace std;

int factorial(int n) {
    // Base case
    if (n <= 1) {
        return 1;
    }
    // Recursive case
    return n * factorial(n - 1);  // Returns result of recursion
}

int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

void countdown(int n) {
    if (n <= 0) {
        cout << "Blast off!" << endl;
        return;  // Base case return
    }
    cout << n << "... ";
    countdown(n - 1);
}

int main() {
    cout << "Factorial of 5: " << factorial(5) << endl;
    cout << "Fibonacci(10): " << fibonacci(10) << endl;
    cout << "Countdown: ";
    countdown(5);
    
    return 0;
}
```

## Practical Examples

### Example 1: Mathematical Functions
```cpp
#include <iostream>
#include <cmath>
using namespace std;

double power(double base, int exponent) {
    if (exponent == 0) return 1;
    if (exponent == 1) return base;
    
    double result = 1;
    for (int i = 0; i < abs(exponent); i++) {
        result *= base;
    }
    
    if (exponent < 0) {
        return 1.0 / result;
    }
    return result;
}

int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int lcm(int a, int b) {
    return (a * b) / gcd(a, b);
}

int main() {
    cout << "2^3 = " << power(2, 3) << endl;
    cout << "2^-2 = " << power(2, -2) << endl;
    cout << "GCD of 12 and 18: " << gcd(12, 18) << endl;
    cout << "LCM of 12 and 18: " << lcm(12, 18) << endl;
    
    return 0;
}
```

### Example 2: String Processing Functions
```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

string reverse(const string& text) {
    string result = text;
    int n = result.length();
    for (int i = 0; i < n / 2; i++) {
        char temp = result[i];
        result[i] = result[n - 1 - i];
        result[n - 1 - i] = temp;
    }
    return result;
}

int countVowels(const string& text) {
    int count = 0;
    for (char c : text) {
        c = tolower(c);
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            count++;
        }
    }
    return count;
}

bool isPalindrome(const string& text) {
    string cleaned;
    for (char c : text) {
        if (isalnum(c)) {
            cleaned += tolower(c);
        }
    }
    
    int left = 0, right = cleaned.length() - 1;
    while (left < right) {
        if (cleaned[left] != cleaned[right]) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}

int main() {
    string word = "Hello World";
    cout << "Original: " << word << endl;
    cout << "Reversed: " << reverse(word) << endl;
    cout << "Vowels: " << countVowels(word) << endl;
    cout << "Is 'racecar' palindrome? " << boolalpha << isPalindrome("racecar") << endl;
    
    return 0;
}
```

### Example 3: Array Processing Functions
```cpp
#include <iostream>
using namespace std;

int sumArray(const int arr[], int size) {
    int total = 0;
    for (int i = 0; i < size; i++) {
        total += arr[i];
    }
    return total;
}

double averageArray(const int arr[], int size) {
    if (size == 0) return 0;
    return static_cast<double>(sumArray(arr, size)) / size;
}

int findMax(const int arr[], int size) {
    if (size == 0) return 0;
    
    int max = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

int findMin(const int arr[], int size) {
    if (size == 0) return 0;
    
    int min = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    return min;
}

int findIndex(const int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return i;
        }
    }
    return -1;
}

int main() {
    int numbers[] = {45, 23, 78, 12, 89, 34};
    int size = sizeof(numbers) / sizeof(numbers[0]);
    
    cout << "Sum: " << sumArray(numbers, size) << endl;
    cout << "Average: " << averageArray(numbers, size) << endl;
    cout << "Max: " << findMax(numbers, size) << endl;
    cout << "Min: " << findMin(numbers, size) << endl;
    cout << "Index of 78: " << findIndex(numbers, size, 78) << endl;
    
    return 0;
}
```

### Example 4: Validation Functions
```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

bool isValidUsername(const string& username) {
    if (username.length() < 3 || username.length() > 20) {
        return false;
    }
    
    for (char c : username) {
        if (!isalnum(c)) {
            return false;
        }
    }
    return true;
}

bool isValidPassword(const string& password) {
    if (password.length() < 8) return false;
    
    bool hasUpper = false, hasLower = false, hasDigit = false;
    for (char c : password) {
        if (isupper(c)) hasUpper = true;
        else if (islower(c)) hasLower = true;
        else if (isdigit(c)) hasDigit = true;
    }
    
    return hasUpper && hasLower && hasDigit;
}

bool isValidEmail(const string& email) {
    size_t atPos = email.find('@');
    size_t dotPos = email.find('.', atPos + 2);
    
    if (atPos == string::npos) return false;
    if (dotPos == string::npos) return false;
    if (atPos == 0) return false;
    if (dotPos == email.length() - 1) return false;
    
    return true;
}

int main() {
    cout << boolalpha;
    
    cout << "Username 'john123': " << isValidUsername("john123") << endl;
    cout << "Username 'jo': " << isValidUsername("jo") << endl;
    
    cout << "Password 'Pass123!': " << isValidPassword("Pass123!") << endl;
    cout << "Password 'password': " << isValidPassword("password") << endl;
    
    cout << "Email 'test@test.com': " << isValidEmail("test@test.com") << endl;
    cout << "Email 'invalid': " << isValidEmail("invalid") << endl;
    
    return 0;
}
```

## Common Mistakes

### Mistake 1: Missing Return in Non-void Function
```cpp
// WRONG - missing return
int add(int a, int b) {
    int sum = a + b;
    // No return! Undefined behavior
}

// CORRECT
int add(int a, int b) {
    int sum = a + b;
    return sum;
}
```

### Mistake 2: Returning Wrong Type
```cpp
// WRONG - returning double from int function
int getNumber() {
    return 3.14;  // Truncated to 3, warning
}

// CORRECT
double getNumber() {
    return 3.14;
}
```

### Mistake 3: Unreachable Code After return
```cpp
// WRONG - code after return never executes
int getValue() {
    return 42;
    cout << "This never prints!" << endl;  // Unreachable
}

// CORRECT
int getValue() {
    cout << "Getting value..." << endl;
    return 42;
}
```

### Mistake 4: Not Handling All Paths
```cpp
// WRONG - missing return for some paths
int absolute(int x) {
    if (x < 0) {
        return -x;
    }
    // What if x >= 0? Missing return!
}

// CORRECT
int absolute(int x) {
    if (x < 0) {
        return -x;
    }
    return x;  // Handles all cases
}
```

### Mistake 5: Returning Local Variable Reference
```cpp
// WRONG - returning reference to local variable
int& getValue() {
    int x = 5;
    return x;  // x destroyed after function ends!
}

// CORRECT - return by value
int getValue() {
    int x = 5;
    return x;
}

// Or use static
int& getStaticValue() {
    static int x = 5;  // Static - exists for program lifetime
    return x;
}
```

## Quick Reference

| Return Type | Syntax | Example |
|-------------|--------|---------|
| void | `return;` | `return;` |
| int | `return value;` | `return 42;` |
| double | `return value;` | `return 3.14;` |
| bool | `return condition;` | `return x > 0;` |
| string | `return expression;` | `return "Hello";` |
| Reference | `return variable;` | `return x;` |
| Pointer | `return ptr;` | `return &x;` |

## Summary

- **return** exits the function and optionally returns a value
- **void** functions can use `return;` (optional)
- Non-void functions must return a value of the correct type
- **Multiple return** statements can exist in a function
- **Early return** can simplify logic and improve performance
- Return **expressions** are evaluated before returning
- **Local variables** are destroyed after return
- Returning **references** to local variables is dangerous
- Return value can be **ignored** in calling code
- **Recursive** functions rely on return for base cases
